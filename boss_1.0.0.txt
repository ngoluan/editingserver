import { NextResponse } from "next/server";

const FRAME_ANCESTORS = [
  "'self'",
  "https://teams.microsoft.com",
  "https://*.teams.microsoft.com",
  "https://*.cloud.microsoft",
  "https://*.office.com",
  "https://outlook.office.com",
  "https://outlook.office365.com",
  "https://*.microsoft365.com",

  // TacoTaco site
  "https://tacotacoto.ca",
  "https://*.tacotacoto.ca",

  // EatTaco site
  "https://eattaco.ca",
  "https://*.eattaco.ca",
].join(" ");

const suspiciousKeywords = [
  '.env', '.bak', '.php', '.yaml', '.git', '.log',
  '.sh', '.sql', '.zip', '.tar.gz', '.json.bak',
  '/debug', '/webhook', '/.aws'
];

export function middleware(request) {
  const url = request.nextUrl.pathname.toLowerCase();

  if (suspiciousKeywords.some(keyword => url.includes(keyword))) {
    return new NextResponse('Forbidden', { status: 403 });
  }

  const response = NextResponse.next();

  response.headers.set(
    "Content-Security-Policy",
    `frame-ancestors ${FRAME_ANCESTORS};`
  );

  response.headers.delete("X-Frame-Options");

  return response;
}

export const config = {
  matcher: '/((?!api|_next/static|_next/image|favicon.ico).*)',
};

function toTitleCase(segment) {
  return segment
    .split('-')
    .filter(Boolean)
    .map((word) => word.charAt(0).toUpperCase() + word.slice(1))
    .join(' ');
}

function formatDocumentTitle(pageTitle) {
  return pageTitle === APP_NAME ? APP_NAME : `${pageTitle} | ${APP_NAME}`;
}

function getPageTitle(pathname) {
  const explicitTitle = routeTitles[pathname];
  if (explicitTitle) {
    return formatDocumentTitle(explicitTitle);
  }

  const segments = pathname.split('/').filter(Boolean);
  const lastStaticSegment = [...segments].reverse().find((segment) => !segment.startsWith('['));

  if (!lastStaticSegment) {
    return APP_NAME;
  }

  return formatDocumentTitle(toTitleCase(lastStaticSegment));
}

function MyApp({ Component, pageProps }) {

import React, { createContext, useState, useEffect, useCallback } from 'react';
import { useRouter } from 'next/router';
import { toast } from 'react-toastify';
import { getOrganizationEmailAccounts } from '@/services/email/utils/api';
import { resolveFetchUrl } from '@/shared/utils/fetchWithAuth';

export const AuthContext = createContext();

const sessionsKey = 'ws.userSessions';
const activeSessionKeyStorage = 'ws.activeSessionKey';
const legacyTokenStorageKey = 'token';

const getSessionKeyForUser = (user) => `${user.id}:${user.organizationId}`;

const isBrowser = () => typeof window !== 'undefined';

const getCurrentEmbeddedValue = () => {
  if (!isBrowser()) return '';

  try {
    const params = new URLSearchParams(window.location.search);
    const embedded = params.get('embedded');
    return typeof embedded === 'string' ? embedded.trim() : '';
  } catch (error) {
    console.error('Failed to read embedded query param', error);
    return '';
  }
};

const appendEmbeddedParam = (path, embeddedValue = getCurrentEmbeddedValue()) => {
  if (!path || !embeddedValue || !isBrowser()) return path;

  try {
    const url = new URL(path, window.location.origin);
    url.searchParams.set('embedded', embeddedValue);
    return `${url.pathname}${url.search}${url.hash}`;
  } catch (error) {
    console.error('Failed to append embedded query param', error);
    return path;
  }
};

const getServerUrl = () => resolveFetchUrl(process.env.NEXT_PUBLIC_SERVER_URL || '');

const readStoredSessions = () => {
  if (!isBrowser()) return {};

  try {
    const raw = localStorage.getItem(sessionsKey);
    if (!raw) return {};
    const parsed = JSON.parse(raw);
    return parsed && typeof parsed === 'object' && !Array.isArray(parsed) ? parsed : {};
  } catch (error) {
    console.error('Failed to read stored user sessions', error);
    return {};
  }
};

const writeStoredSessions = (sessions) => {
  if (!isBrowser()) return;

  try {
    localStorage.setItem(sessionsKey, JSON.stringify(sessions || {}));
  } catch (error) {
    console.error('Failed to write stored user sessions', error);
  }
};

const syncLegacyActiveToken = (sessionsOverride = null) => {
  if (!isBrowser()) return '';

  try {
    const sessions = sessionsOverride || readStoredSessions();
    const activeKey = localStorage.getItem(activeSessionKeyStorage) || '';
    const activeSession = activeKey ? sessions[activeKey] : null;

    const fallbackSession =
      (activeSession && typeof activeSession.token === 'string' && activeSession.token.trim() ? activeSession : null) ||
      Object.values(sessions).find(
        (session) => session && typeof session.token === 'string' && session.token.trim()
      ) ||
      null;

    const token = fallbackSession?.token?.trim() || '';

    if (fallbackSession?.key && activeKey !== fallbackSession.key) {
      localStorage.setItem(activeSessionKeyStorage, fallbackSession.key);
    }

    if (token) {
      localStorage.setItem(legacyTokenStorageKey, token);
    } else {
      localStorage.removeItem(legacyTokenStorageKey);
    }

    return token;
  } catch (error) {
    console.error('Failed to sync legacy token', error);
    return '';
  }
};

const removeStoredSession = (key) => {
  if (!isBrowser()) return;

  try {
    const sessions = readStoredSessions();
    if (sessions[key]) {
      delete sessions[key];
      writeStoredSessions(sessions);
      syncLegacyActiveToken(sessions);
    }

    if (localStorage.getItem(activeSessionKeyStorage) === key) {
      localStorage.removeItem(activeSessionKeyStorage);
      syncLegacyActiveToken(sessions);
    }
  } catch (error) {
    console.error('Failed to remove stored session', error);
  }
};

const clearStoredSessions = () => {
  if (!isBrowser()) return;

  try {
    localStorage.removeItem(sessionsKey);
    localStorage.removeItem(activeSessionKeyStorage);
    localStorage.removeItem(legacyTokenStorageKey);
  } catch (error) {
    console.error('Failed to clear stored sessions', error);
  }
};

const upsertUserSession = (sessionData) => {
  if (!isBrowser()) return;

  try {
    if (!sessionData?.user) return;

    const sessionKey = sessionData.key || getSessionKeyForUser(sessionData.user);
    const sessions = readStoredSessions();
    const existing = sessions[sessionKey] || {};

    const nextSession = {
      key: sessionKey,
      token:
        typeof sessionData.token === 'string' && sessionData.token.trim()
          ? sessionData.token.trim()
          : existing.token || '',
      userId: sessionData.user.id,
      username: sessionData.user.username,
      organizationId: sessionData.user.organizationId,
      organizationName: sessionData.user.organizationName,
      organizationSlug: sessionData.user.organizationSlug || '',
      permissions: sessionData.user.permissions || [],
      lastLoginAt: Date.now(),
    };

    sessions[sessionKey] = nextSession;

    writeStoredSessions(sessions);
    localStorage.setItem(activeSessionKeyStorage, sessionKey);
    syncLegacyActiveToken(sessions);
  } catch (error) {
    console.error('Failed to update user sessions in localStorage', error);
  }
};

export const AuthProvider = ({ children }) => {
  const [user, setUser] = useState(null);
  const [organization, setOrganization] = useState(null);

  const [loading, setLoading] = useState(true);

        return token;

      const fallbackToken = syncLegacyActiveToken(sessions);
      if (fallbackToken) return fallbackToken;

      const legacyToken = localStorage.getItem(legacyTokenStorageKey);
      return typeof legacyToken === 'string' ? legacyToken.trim() : '';

        setOrganization(null);

    upsertUserSession({ ...data, key: sessionKey });

  };

  return (
    <AuthContext.Provider
      value={{

        permissions: user ? user.permissions : [],
        organization,

import React, { createContext, useContext, useEffect, useState } from 'react';
import io from 'socket.io-client';
import { AuthContext } from './AuthContext';

// FIX: Initialize with a safe default object so destructuring doesn't crash the app
// if the Provider is missing or loading.
const SocketContext = createContext({ socket: null, connected: false });

export const useSocket = () => {
  return useContext(SocketContext);
};

export const SocketProvider = ({ children }) => {
  const { user } = useContext(AuthContext);
  const [socket, setSocket] = useState(null);
  const [connected, setConnected] = useState(false);

  useEffect(() => {
    if (!user) return;

    const newSocket = io(process.env.NEXT_PUBLIC_SERVER_URL, {
      auth: {
        token: user.token,
      },
      transports: ['websocket'],
    });

    newSocket.on('connect', () => {
      console.log('Connected to Socket.IO server');
      setConnected(true);
    });

    newSocket.on('disconnect', () => {
      console.log('Disconnected from Socket.IO server');
      setConnected(false);
    });

    newSocket.on('connect_error', (err) => {
      console.error('Socket connection error:', err.message);
    });

    setSocket(newSocket);

    return () => {
      newSocket.disconnect();
    };
  }, [user]);

  return (
    <SocketContext.Provider value={{ socket, connected }}>
      {children}
    </SocketContext.Provider>
  );
};

import { toast } from 'react-toastify';

const sockets = {}; // Keep track of sockets per namespace

export const useGenericSocket = (namespace) => {
  const [socket, setSocket] = useState(null);
  const [connected, setConnected] = useState(false);
  const namespaceRef = useRef(namespace); // Store namespace in ref

  useEffect(() => {
    const currentNamespace = namespaceRef.current; // Use ref value

    // Only initialize if namespace is valid and socket doesn't exist or is disconnected
    if (currentNamespace && (!sockets[currentNamespace] || !sockets[currentNamespace].connected)) {
       // Disconnect previous instance if it exists and is different namespace
       if (socket && socket.nsp !== `/${currentNamespace}`) {
          socket.disconnect();
       }

      console.log(`Initializing socket for namespace: /${currentNamespace}`);
      sockets[currentNamespace] = io(`${process.env.NEXT_PUBLIC_SERVER_URL}/${currentNamespace}`, {
        withCredentials: true,
        transports: ['websocket'],
        // Consider adding autoConnect: false and connecting manually if needed
      });

      setSocket(sockets[currentNamespace]); // Update state
    } else if (sockets[currentNamespace]) {
         setSocket(sockets[currentNamespace]); // Use existing socket
         setConnected(sockets[currentNamespace].connected);
    }

    const currentSocket = sockets[currentNamespace];

    if (!currentSocket) return; // Exit if no socket could be established

    const handleConnect = () => {
      setConnected(true);
      console.log(`Connected to /${currentNamespace} namespace`);
    };
    const handleDisconnect = () => {
      setConnected(false);
      console.log(`Disconnected from /${currentNamespace} namespace`);
      // Optionally attempt reconnection or notify user
    };
    const handleConnectError = (err) => {
      console.error(`${currentNamespace} Socket connection error:`, err.message);
      // Avoid toast for every error, maybe only specific ones or after retries fail
      // toast.error(`${currentNamespace} Socket connection error: ${err.message}`);
    };

     // Remove previous listeners before adding new ones
    currentSocket.off('connect', handleConnect);
    currentSocket.off('disconnect', handleDisconnect);
    currentSocket.off('connect_error', handleConnectError);

    currentSocket.on('connect', handleConnect);
    currentSocket.on('disconnect', handleDisconnect);
    currentSocket.on('connect_error', handleConnectError);

     // If socket is already connected when effect runs
    if (currentSocket.connected) {
      handleConnect();
    }

    // Connect if not already connected (useful with autoConnect: false)
    // if (!currentSocket.connected) {
    //   currentSocket.connect();
    // }

    return () => {
      console.log(`Cleaning up socket listeners for /${currentNamespace}`);
      // Only disconnect if the component using this specific instance unmounts
      // Global sockets might be managed differently (e.g., singleton pattern)
      // For simplicity here, we'll keep it, but review if sockets should persist globally
        // currentSocket.off('connect', handleConnect);
        // currentSocket.off('disconnect', handleDisconnect);
        // currentSocket.off('connect_error', handleConnectError);
      // Consider if disconnecting here is always desired.
      // If you want the socket to persist across component mounts/unmounts,
      // you might remove the disconnect() call or manage it globally.
      // currentSocket.disconnect();
      // delete sockets[currentNamespace]; // Clean up if disconnecting
    };
  }, [namespace]); // Depend only on the namespace string

  // Return the current socket instance and connection status
  return { socket, connected };
};

export const NATIVE_IOS_PLATFORM = 'native-ios';

export const getClientPlatform = () => {
  if (typeof window === 'undefined') return 'web';

  const nativeBridge = window.WorksmrtNative;
  if (nativeBridge?.isNative && nativeBridge?.platform === 'ios') {
    return NATIVE_IOS_PLATFORM;
  }
  if (nativeBridge?.isNative && nativeBridge?.platform === 'android') {
    return 'native-android';
  }

  const capacitor = window.Capacitor;
  const capacitorPlatform =
    typeof capacitor?.getPlatform === 'function' ? capacitor.getPlatform() : null;
  const isNative =
    typeof capacitor?.isNativePlatform === 'function'
      ? capacitor.isNativePlatform()
      : capacitor?.isNative === true || Boolean(capacitor?.Plugins?.WorksmrtNative);

  if (isNative && capacitorPlatform === 'ios') return NATIVE_IOS_PLATFORM;
  if (isNative && capacitorPlatform === 'android') return 'native-android';

  return 'web';
};

export const isNativeIOS = () => getClientPlatform() === NATIVE_IOS_PLATFORM;

export const buildClientPlatformHeaders = () => ({
  [CLIENT_PLATFORM_HEADER]: getClientPlatform(),

// client/shared/utils/fetchWithAuth.js
import { toast } from 'react-toastify';
import { buildClientPlatformHeaders } from '@/shared/utils/clientPlatform';

const sessionsKey = 'ws.userSessions';
const activeSessionKeyStorage = 'ws.activeSessionKey';
const legacyTokenStorageKey = 'token';

const readStoredSessions = () => {
  if (typeof window === 'undefined') return {};

  try {
    const raw = window.localStorage.getItem(sessionsKey);
    if (!raw) return {};
    const parsed = JSON.parse(raw);
    return parsed && typeof parsed === 'object' && !Array.isArray(parsed) ? parsed : {};
  } catch (error) {
    console.error('[fetchWithAuth] Failed to read stored user sessions', error);
    return {};
  }
};

const writeTokenToStorage = (token) => {
  if (typeof window === 'undefined' || !token || typeof token !== 'string') return;

  const trimmedToken = token.trim();
  if (!trimmedToken) return;

  try {
    const sessions = readStoredSessions();
    const activeKey = window.localStorage.getItem(activeSessionKeyStorage) || '';

    if (activeKey && sessions[activeKey] && typeof sessions[activeKey] === 'object') {
      sessions[activeKey] = {
        ...sessions[activeKey],
        token: trimmedToken,
      };
      window.localStorage.setItem(sessionsKey, JSON.stringify(sessions));
    }

    window.localStorage.setItem(legacyTokenStorageKey, trimmedToken);
  } catch (error) {
    console.error('[fetchWithAuth] Failed to persist refreshed auth token', error);
  }
};

const clearStoredAuthToken = () => {
  if (typeof window === 'undefined') return;

  try {
    const sessions = readStoredSessions();
    const activeKey = window.localStorage.getItem(activeSessionKeyStorage) || '';

    if (activeKey && sessions[activeKey] && typeof sessions[activeKey] === 'object') {
      delete sessions[activeKey].token;
      window.localStorage.setItem(sessionsKey, JSON.stringify(sessions));
    }

    window.localStorage.removeItem(legacyTokenStorageKey);
  } catch (error) {
    console.error('[fetchWithAuth] Failed to clear stored auth token', error);
  }
};

export const getStoredAuthToken = () => {
  if (typeof window === 'undefined') return '';

  try {
    const sessions = readStoredSessions();
    const activeKey = window.localStorage.getItem(activeSessionKeyStorage) || '';
    const activeSession = activeKey ? sessions[activeKey] : null;

    if (activeSession && typeof activeSession.token === 'string' && activeSession.token.trim()) {
      const token = activeSession.token.trim();
      window.localStorage.setItem(legacyTokenStorageKey, token);
      return token;
    }

    const fallbackSession = Object.values(sessions).find(
      (session) => session && typeof session.token === 'string' && session.token.trim()
    );

    if (fallbackSession?.token) {
      const token = fallbackSession.token.trim();
      window.localStorage.setItem(legacyTokenStorageKey, token);
      return token;
    }

    const legacyToken = window.localStorage.getItem(legacyTokenStorageKey);
    return typeof legacyToken === 'string' ? legacyToken.trim() : '';
  } catch (error) {
    console.error('[fetchWithAuth] Failed to resolve stored auth token', error);
    return '';
  }
};

const buildHeaders = (options = {}, token = '') => {
  const headers = {
    ...buildClientPlatformHeaders(),
    ...(options.headers || {}),
  };

  if (token && !headers.Authorization) {
    headers.Authorization = `Bearer ${token}`;
  }

  return headers;
};

export const resolveFetchUrl = (url) => {
  if (typeof window === 'undefined') return url;

  const configuredUrl = process.env.NEXT_PUBLIC_SERVER_URL || '';
  if (!configuredUrl) return url;

  try {
    const apiUrl = new URL(configuredUrl, window.location.origin);
    const requestUrl = new URL(String(url), window.location.origin);
    const localApiHosts = ['localhost', '127.0.0.1'];
    const isLocalApi =
      localApiHosts.includes(apiUrl.hostname) && ['3030', '3031'].includes(apiUrl.port);
    const isLocalPage = localApiHosts.includes(window.location.hostname);

    if (!isLocalApi || isLocalPage || requestUrl.origin !== apiUrl.origin) {
      return url;
    }

    const pathname = requestUrl.pathname === '/' ? '' : requestUrl.pathname;
    return `/api/server${pathname}${requestUrl.search}${requestUrl.hash}`;
  } catch {
    return url;
  }
};

const doFetch = async (url, options, headers) => {
  return fetch(resolveFetchUrl(url), {
    ...options,
    credentials: 'include',
    headers,
  });
};

const tryRefreshSessionToken = async () => {
  if (typeof window === 'undefined') return '';

  try {
    const baseUrl = process.env.NEXT_PUBLIC_SERVER_URL;
    if (!baseUrl) return '';

    const response = await fetch(resolveFetchUrl(`${baseUrl}/user/session/current`), {
      method: 'GET',
      credentials: 'include',
    });

    if (!response.ok) {
      return '';
    }

    const payload = await response.json().catch(() => null);
    const refreshedToken = typeof payload?.token === 'string' ? payload.token.trim() : '';

    if (refreshedToken) {
      writeTokenToStorage(refreshedToken);
    }

    return refreshedToken;
  } catch (error) {
    console.error('[fetchWithAuth] Session refresh failed', error);
    return '';
  }
};

export async function fetchWithAuth(url, options = {}) {
  if (!url) {
    throw new Error('fetchWithAuth requires a URL');
  }

  const { silent = false, ...fetchOptions } = options;
  options = fetchOptions;

  const initialToken = getStoredAuthToken();
  let headers = buildHeaders(options, initialToken);

  try {
    console.log('[fetchWithAuth] Request ->', url, {
      ...options,
      headers,
    });

    let response = await doFetch(url, options, headers);

    if (response.status === 401 && headers.Authorization) {
      const cookieOnlyHeaders = { ...(options.headers || {}) };
      response = await doFetch(url, options, cookieOnlyHeaders);

      if (response.ok) {
        clearStoredAuthToken();
        return response;
      }

      const refreshedToken = await tryRefreshSessionToken();
      if (refreshedToken) {
        headers = buildHeaders(options, refreshedToken);
        response = await doFetch(url, options, headers);
        if (response.ok) {
          return response;
        }
      }
    }

    if (response.status === 401) {
      if (!silent) toast.error('Your session has expired. Please log in again.');
    }

    return response;
  } catch (error) {
    if (error?.name === 'AbortError') {
      const wrappedError = new Error('Request was cancelled.');
      wrappedError.cause = error;
      throw wrappedError;
    }

    const message = `Failed to reach server: ${url}`;
    console.error('[fetchWithAuth] Error:', { url, options, error });
    if (!silent) toast.error(message);

    const wrappedError = new Error(message);
    wrappedError.cause = error;
    throw wrappedError;
  }
}

import { getSession } from '#shared/utils/auth.js';
import { extractBearerToken } from '#shared/utils/bearerToken.js';
import jwt from 'jsonwebtoken';

const buildRequestUser = (decoded = {}) => {
  const normalizedUserId = decoded.userId || decoded.id || null;

  return {
    ...decoded,
    id: normalizedUserId,
    userId: normalizedUserId,
  };
};

const authenticate = async (req, res, next) => {
  if (req.headers['x-internal-request'] === 'true') {
    // For internal requests, set a minimal user context if none exists
    if (!req.user) {
      req.user = { organizationId: null, id: null, role: 'INTERNAL' };
    }
    return next();
  }

  const authHeader = req.headers.authorization;
  const hasBearerToken = authHeader && /^Bearer\s+/i.test(authHeader);

  if (hasBearerToken) {
    const token = extractBearerToken(authHeader);

    try {
      if (!process.env.JWT_SECRET) {
        console.error('JWT_SECRET is not defined in environment variables.');
        return res.status(500).json({ error: 'Server Configuration Error' });
      }

      const decoded = jwt.verify(token, process.env.JWT_SECRET);
      req.user = buildRequestUser(decoded);
      return next();
    } catch (err) {
      console.warn(`[Auth] JWT verification failed, attempting cookie/session fallback: ${err.message}`);
    }
  }

  const session = await getSession(req);
  if (session?.user) {
    req.user = session.user;
    req.session = session;
    return next();
  }

  if (hasBearerToken) {
    return res.status(401).json({
      error: 'Invalid or Expired Token',
      needsReauth: true,
    });
  }

  return res.status(401).json({ error: 'Unauthorized' });
};

export default authenticate;

import { extractRequestContext, logAndReportError } from '../utils/logError.js';

export const errorHandler = (err, req, res, next) => {
  logAndReportError('Error:', err, extractRequestContext(req));

  if (err.code && err.code.startsWith('P')) {
    switch (err.code) {
      case 'P2002':
        return res.status(409).json({ error: 'Unique constraint failed on the fields: ' + err.meta.target });
      case 'P2003':
        return res.status(400).json({ error: 'Foreign key constraint failed on the field: ' + err.meta.field_name });
      default:
        return res.status(500).json({ error: 'Database error occurred.' });
    }
  }

  const statusCode = err.status || 500;
  res.status(statusCode).json({
    error: err.message || 'Internal Server Error',
  });
};

export const sendErrorResponse = (res, error, context = null) => {
  logAndReportError('Error:', error, context);
  const statusCode = error.status || 500;
  res.status(statusCode).json({
    error: error.message || 'Internal Server Error',
  });
};

    if (forceShutdownTimer) {
      clearTimeout(forceShutdownTimer);
      forceShutdownTimer = null;
    }

    await flushConsoleCaptureWrites();
    process.exit(0);
  } catch (error) {

      console.log('Prisma disconnected after shutdown error');
    } catch (disconnectError) {
      console.error('Failed to disconnect Prisma after shutdown error:', disconnectError);
    }

    if (forceShutdownTimer) {
      clearTimeout(forceShutdownTimer);
      forceShutdownTimer = null;
    }

    await flushConsoleCaptureWrites();
    process.exit(1);
  }
};

/**
 * A higher-order function to wrap asynchronous route handlers.
 * It catches any errors and passes them to the next middleware.
 *
 * @param {Function} fn - The asynchronous route handler function.
 * @returns {Function} A new function that wraps the original handler.
 */
const asyncHandler = (fn) => {
    // Ensure that fn is a function
    if (typeof fn !== 'function') {
      throw new TypeError('Expected a function as the argument');
    }
  
    // Return a new function that wraps the original handler
    return function asyncUtilWrap(req, res, next) {
      // Call the handler and catch any errors
      Promise.resolve(fn(req, res, next)).catch((error) => {
        logAndReportError('Async route handler error:', error, extractRequestContext(req));
        next(error);
      });
    };
  };
  
  export default asyncHandler;

import { parse } from 'cookie';
import jwt from 'jsonwebtoken';
import prisma from '#shared/database/prismaClient.js';

const JWT_SECRET = process.env.JWT_SECRET || 'your_jwt_secret_here';

export const getTokenFromRequest = (req) => {
  const cookies = req.headers.cookie ? parse(req.headers.cookie) : {};
  const cookieToken = cookies.token;

  if (cookieToken) {
    return cookieToken;
  }

  const authHeader = req.headers.authorization;
  if (typeof authHeader === 'string' && authHeader.startsWith('Bearer ')) {
    const bearerToken = authHeader.slice('Bearer '.length).trim();
    return bearerToken || null;
  }

  return null;
};

export async function getSession(req) {
  try {
    const token = getTokenFromRequest(req);

    if (!token) {
      console.log('No auth token found in cookies or Authorization header.');
      return null;
    }

    const decoded = jwt.verify(token, JWT_SECRET);
    const normalizedUserId = decoded?.userId || decoded?.id || null;

    if (!normalizedUserId) {
      console.warn('Session token missing userId/id.', {
        sessionId: decoded?.sessionId || null,
        username: decoded?.username || null,
      });
      return null;
    }

    const user = await prisma.user.findUnique({
      where: { id: normalizedUserId },
      include: {
        organization: {
          select: { id: true, name: true, phoneNumber: true, slug: true },
        },
        permissions: {
          select: { name: true },
        },
      },
    });

    if (!user) {
      console.log('User not found in database.');
      return null;
    }

    if (user.deletedAt || !user.active) {
      console.log('User account has been deactivated or deleted:', user.id);
      return null;
    }

    if (!user.organization) {
      console.log('Organization not found for user:', user.id);
      return null;
    }

    const sessionUser = {
      ...user,
      id: user.id,
      userId: user.id,
      organizationId: user.organization.id,
      organizationName: user.organization.name,
      organizationSlug: user.organization.slug,
      permissions: user.permissions.map((p) => p.name),
      sessionId: decoded?.sessionId || null,
    };

    return { user: sessionUser };
  } catch (error) {
    console.error('Session Error:', error);
    return null;
  }
}

export function extractBearerToken(authHeader = "") {
  const rawHeader = String(authHeader || "").trim();
  const match = rawHeader.match(/^Bearer\s+([\s\S]+)$/i);
  if (!match) return "";

  let token = match[1].trim();
  token = token.replace(/^["'`]+|["'`]+$/g, "");

  if (token.startsWith("<") && token.endsWith(">")) {
    token = token.slice(1, -1).trim();
  }

  return token;
}

export function isLikelyJwt(value = "") {
  return /^[A-Za-z0-9_-]+\.[A-Za-z0-9_-]+\.[A-Za-z0-9_-]+$/.test(String(value || "").trim());
}

// Full path: /home/luanngo/boss/server/src/shared/utils/withHandler.js
import { getSession } from './auth.js';
import { extractBearerToken } from './bearerToken.js';
import { sendErrorResponse } from '#middleware/errorHandler.js';
import { extractRequestContext, logAndReportError } from '../../utils/logError.js';
import jwt from 'jsonwebtoken';

/**
 * Higher-Order Function to handle API route common logic.
 * Updated to support Bearer Token authentication for M2M communication.
 *
 * @param {Object} options - Configuration options.
 * @param {Function} options.handler - The main handler function for the route.
 * @param {Array} options.allowedMethods - List of allowed HTTP methods.
 * @param {Array} [options.requiredRoles=[]] - List of roles required to access the route.
 * @param {Object} [options.schema={}] - Object containing Zod schemas for different methods.
 * @param {Object} [options.requiredBodyParams={}] - Required body parameters per method.
 * @param {Object} [options.requiredQueryParams={}] - Required query parameters per method.
 * @param {Object} [options.requiredBodyParamTypes={}] - Expected types for body parameters per method.
 * @param {Object} [options.requiredQueryParamTypes={}] - Expected types for query parameters per method.
 * @param {boolean} [options.requiresAuth=true] - Whether authentication is required.
 *
 * @returns {Function} - Wrapped API route handler.
 */
const withHandler = ({
  handler,
  allowedMethods = [],
  requiredRoles = [],
  schema = {},
  requiredBodyParams = {},
  requiredQueryParams = {},
  requiredBodyParamTypes = {},
  requiredQueryParamTypes = {},
  requiresAuth = true,
}) => {
  return async (req, res) => {
    try {
      const { method } = req;

      // **1. Method Validation**
      if (!allowedMethods.includes(method)) {
        res.setHeader('Allow', allowedMethods);
        return res.status(405).end(`Method ${method} Not Allowed`);
      }

      // **2. Authentication (Session OR Bearer Token)**
      if (requiresAuth) {
        let user = null;

        // A. Check for Bearer Token (API/M2M Access)
        const authHeader = req.headers.authorization;
        if (authHeader && /^Bearer\s+/i.test(authHeader)) {
          const token = extractBearerToken(authHeader);
          try {
            const secret = process.env.JWT_SECRET;
            if (!secret) throw new Error('JWT_SECRET not configured');
            
            const decoded = jwt.verify(token, secret);
            const normalizedUserId = decoded?.id || decoded?.userId || null;
            user = {
              ...decoded,
              ...(normalizedUserId ? { id: normalizedUserId, userId: normalizedUserId } : {}),
            };
          } catch (err) {
            console.warn(`[Auth] Invalid Bearer token: ${err.message}`);
            // We continue to check session as a fallback, or fail here.
          }
        }

        // B. Check for Cookie Session (Browser Access)
        if (!user) {
          const session = await getSession(req);
          if (session && session.user) {
            user = session.user;
          }
        }

        // C. Final Auth Check
        if (!user) {
          return res.status(401).json({ error: 'Unauthorized: No valid session or token found.' });
        }

        req.user = user;
        
        // Implement permissions check here if needed based on requiredRoles
        if (requiredRoles.length > 0) {
            // Example simple role check
            if (!req.user.role || !requiredRoles.includes(req.user.role)) {
                return res.status(403).json({ error: 'Forbidden: Insufficient permissions' });
            }
        }
      }

      // **3. Schema Validation with Zod**
      const methodSchema = schema[method];
      if (methodSchema) {
        // Determine which part to validate based on HTTP method
        let validationTarget = method === 'GET' ? req.query : req.body;
        const validation = methodSchema.safeParse(validationTarget);

        if (!validation.success) {
          // Extract and format Zod validation errors
          const errorMessages = validation.error.errors.map(err => ({
            path: err.path.join('.'),
            message: err.message,
          }));
          console.log('Validation Errors:', errorMessages);
          return res.status(400).json({ error: errorMessages });
        }

        // Attach validated data to the request object for use in the handler
        req.validated = validation.data;
      } else {
        // **4. Existing Manual Validation (Legacy support)**
        // Validate required body parameters
        const bodyParams = requiredBodyParams[method] || [];
        if (bodyParams.length > 0) {
          const missingParams = bodyParams.filter(
            (param) => !(param in req.body)
          );
          if (missingParams.length > 0) {
            return res.status(400).json({
              error: `Missing required fields: ${missingParams.join(', ')}`,
            });
          }
        }

        // Validate required query parameters
        const queryParams = requiredQueryParams[method] || [];
        if (queryParams.length > 0) {
          const missingParams = queryParams.filter(
            (param) => !(param in req.query)
          );
          if (missingParams.length > 0) {
            return res.status(400).json({
              error: `Missing required query parameters: ${missingParams.join(
                ', '
              )}`,
            });
          }
        }

        // **Type Validation for Body Params**
        const bodyTypes = requiredBodyParamTypes[method] || {};
        for (const [param, type] of Object.entries(bodyTypes)) {
          if (type === 'string' && typeof req.body[param] !== 'string') {
            return res.status(400).json({ error: `${param} must be a string.` });
          }
          if (type === 'number' && typeof req.body[param] !== 'number') {
            return res.status(400).json({ error: `${param} must be a number.` });
          }
          if (type === 'boolean' && typeof req.body[param] !== 'boolean') {
            return res.status(400).json({ error: `${param} must be a boolean.` });
          }
        }

        // **Type Validation for Query Params**
        const queryTypes = requiredQueryParamTypes[method] || {};
        for (const [param, type] of Object.entries(queryTypes)) {
          if (type === 'string' && typeof req.query[param] !== 'string') {
            return res.status(400).json({ error: `${param} must be a string.` });
          }
          if (type === 'number' && isNaN(Number(req.query[param]))) {
            return res.status(400).json({ error: `${param} must be a number.` });
          }
          if (type === 'boolean') {
            const val = req.query[param].toLowerCase();
            if (val !== 'true' && val !== 'false') {
              return res.status(400).json({ error: `${param} must be a boolean.` });
            }
          }
        }
      }

      // **5. Execute the Main Handler**
      return handler(req, res);
    } catch (error) {
      logAndReportError('withHandler error:', error, extractRequestContext(req));
      sendErrorResponse(res, error, extractRequestContext(req));
    }
  };
};

export default withHandler;
