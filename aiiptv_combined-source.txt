const fs = require('fs');
const path = require('path');
const os = require('os');
const { execFileSync } = require('child_process');
const readline = require('readline');

function exists(p) {
  return fs.existsSync(p);
}

function ensureDir(dirPath) {
  fs.mkdirSync(dirPath, { recursive: true });
}

function hasFile(dir, fileName) {
  return exists(path.join(dir, fileName));
}

function hasAndroidProjectRoot(dir) {
  return (
    (hasFile(dir, 'settings.gradle.kts') || hasFile(dir, 'settings.gradle')) &&
    exists(path.join(dir, 'app', 'src', 'main', 'AndroidManifest.xml')) &&
    (
      exists(path.join(dir, 'app', 'build.gradle.kts')) ||
      exists(path.join(dir, 'app', 'build.gradle'))
    )
  );
}

function hasAndroidAppModule(dir) {
  return (
    exists(path.join(dir, 'src', 'main', 'AndroidManifest.xml')) &&
    (
      hasFile(dir, 'build.gradle.kts') ||
      hasFile(dir, 'build.gradle')
    )
  );
}

function findRealProjectRoot(startDir) {
  let dir = path.resolve(startDir);

  // If the script is accidentally inside the app module,
  // use the parent project root instead.
  if (hasAndroidAppModule(dir)) {
    const parent = path.dirname(dir);

    if (hasAndroidProjectRoot(parent)) {
      return parent;
    }
  }

  // Walk upward until we find the real Android project root.
  while (true) {
    if (hasAndroidProjectRoot(dir)) {
      return dir;
    }

    const parent = path.dirname(dir);

    if (parent === dir) {
      break;
    }

    dir = parent;
  }

  throw new Error(
    `Could not find Android project root from:\n${startDir}\n\n` +
    `Run this script from the folder that contains:\n` +
    `  settings.gradle.kts\n` +
    `  app/src/main/AndroidManifest.xml`
  );
}

const PROJECT_ROOT = findRealProjectRoot(__dirname);
const DOWNLOADS_DIR = path.join(os.homedir(), 'Downloads');
const FIXES_DIR = path.join(PROJECT_ROOT, 'fixes');
const TEMP_EXTRACT_DIR = path.join(PROJECT_ROOT, '.temp_extracted_fixes');
const BACKUP_ROOT = path.join(PROJECT_ROOT, '.backup');

const SKIP_DIR_NAMES = new Set([
  '.git',
  '.gradle',
  '.idea',
  '.vscode',
  'build',
  '.temp_extracted_fixes',
  '.backup',
  'fixes'
]);

const SKIP_FILE_NAMES = new Set([
  'local.properties'
]);

function getLineCount(filePath) {
  try {
    return fs.readFileSync(filePath, 'utf-8').split('\n').length;
  } catch {
    return 0;
  }
}

function psQuote(value) {
  return `'${String(value).replace(/'/g, "''")}'`;
}

function cleanupTempDir() {
  if (exists(TEMP_EXTRACT_DIR)) {
    fs.rmSync(TEMP_EXTRACT_DIR, {
      recursive: true,
      force: true
    });
  }
}

function resolveDownloadedZip(zipFileName) {
  const fileNameOnly = path.basename(zipFileName);
  const downloadZipPath = path.join(DOWNLOADS_DIR, fileNameOnly);

  if (!exists(downloadZipPath)) {
    throw new Error(
      `ZIP not found in Downloads.\n\n` +
      `Expected:\n${downloadZipPath}\n\n` +
      `Move the downloaded ZIP there, then run:\n` +
      `node apply-fixes.js apply ${fileNameOnly}`
    );
  }

  return downloadZipPath;
}

function stageZipFromDownloads(zipFileName) {
  const sourceZipPath = resolveDownloadedZip(zipFileName);
  ensureDir(FIXES_DIR);

  const stagedZipPath = path.join(FIXES_DIR, path.basename(zipFileName));
  fs.copyFileSync(sourceZipPath, stagedZipPath);

  console.log(`Copied ZIP from: ${sourceZipPath}`);
  console.log(`Copied ZIP to:   ${stagedZipPath}`);

  return stagedZipPath;
}

function expandZip(zipPath, destinationPath) {
  ensureDir(destinationPath);

  if (process.platform === 'win32') {
    const command =
      `Expand-Archive -LiteralPath ${psQuote(zipPath)} ` +
      `-DestinationPath ${psQuote(destinationPath)} -Force`;

    execFileSync(
      'powershell.exe',
      [
        '-NoProfile',
        '-ExecutionPolicy',
        'Bypass',
        '-Command',
        command
      ],
      {
        stdio: 'inherit'
      }
    );

    return;
  }

  try {
    execFileSync(
      'unzip',
      ['-q', '-o', zipPath, '-d', destinationPath],
      {
        stdio: 'inherit'
      }
    );
  } catch {
    execFileSync(
      'tar',
      ['-xf', zipPath, '-C', destinationPath],
      {
        stdio: 'inherit'
      }
    );
  }
}

function walkDirs(startDir, out = []) {
  if (!exists(startDir)) {
    return out;
  }

  const entries = fs.readdirSync(startDir, {
    withFileTypes: true
  });

  for (const entry of entries) {
    const fullPath = path.join(startDir, entry.name);

    if (entry.isDirectory()) {
      if (SKIP_DIR_NAMES.has(entry.name)) {
        continue;
      }

      out.push(fullPath);
      walkDirs(fullPath, out);
    }
  }

  return out;
}

function isAndroidProjectRoot(dir) {
  return hasAndroidProjectRoot(dir);
}

function isAndroidAppFolder(dir) {
  return hasAndroidAppModule(dir);
}

function findExtractedProjectRoot(startDir) {
  const dirs = [startDir, ...walkDirs(startDir)];

  // Prefer a full Android project root.
  for (const dir of dirs) {
    if (isAndroidProjectRoot(dir)) {
      return dir;
    }
  }

  // Fallback: if the ZIP only contains the app module,
  // return the app module itself.
  for (const dir of dirs) {
    if (isAndroidAppFolder(dir)) {
      return dir;
    }
  }

  throw new Error(
    `Could not find Android project root inside extracted ZIP.\n\n` +
    `Expected one of these structures:\n\n` +
    `Full project:\n` +
    `  settings.gradle.kts\n` +
    `  app/src/main/AndroidManifest.xml\n\n` +
    `App-only patch:\n` +
    `  src/main/AndroidManifest.xml\n` +
    `  build.gradle.kts`
  );
}

function shouldSkipPath(relativePath) {
  const parts = relativePath.split(path.sep);

  if (parts.some(part => SKIP_DIR_NAMES.has(part))) {
    return true;
  }

  const fileName = path.basename(relativePath);

  if (SKIP_FILE_NAMES.has(fileName)) {
    return true;
  }

  return false;
}

function collectFiles(dir, baseDir = dir, out = []) {
  const entries = fs.readdirSync(dir, {
    withFileTypes: true
  });

  for (const entry of entries) {
    const fullPath = path.join(dir, entry.name);
    const relativePath = path.relative(baseDir, fullPath);

    if (shouldSkipPath(relativePath)) {
      continue;
    }

    if (entry.isDirectory()) {
      collectFiles(fullPath, baseDir, out);
    } else if (entry.isFile()) {
      out.push({
        fullPath,
        relativePath
      });
    }
  }

  return out;
}

function filesAreEqual(fileA, fileB) {
  if (!exists(fileA) || !exists(fileB)) {
    return false;
  }

  const statA = fs.statSync(fileA);
  const statB = fs.statSync(fileB);

  if (statA.size !== statB.size) {
    return false;
  }

  return fs.readFileSync(fileA).equals(fs.readFileSync(fileB));
}

function parseApplyOptions(args = []) {
  return {
    createNew: !args.includes('--no-create-new')
  };
}

async function askYesNo(question) {
  const rl = readline.createInterface({
    input: process.stdin,
    output: process.stdout
  });

  const answer = await new Promise(resolve => rl.question(question, resolve));
  rl.close();

  return ['y', 'yes'].includes(answer.trim().toLowerCase());
}

function assertSafeTargetRoot() {
  const cwd = path.resolve(process.cwd());
  const projectRoot = path.resolve(PROJECT_ROOT);

  console.log(`Project root: ${projectRoot}`);
  console.log(`Current dir:   ${cwd}`);
  console.log(`Script path:   ${__filename}`);

  // This prevents running from AIIPTVViewer/app.
  if (path.basename(cwd).toLowerCase() === 'app' && hasAndroidAppModule(cwd)) {
    throw new Error(
      `You are running this from inside the Android app module:\n${cwd}\n\n` +
      `This can create a duplicate app/app structure.\n\n` +
      `Run it from the real project root instead:\n` +
      `cd "${projectRoot}"\n` +
      `node apply-fixes.js apply <zip-name>`
    );
  }

  // This prevents PROJECT_ROOT from ever being the app module.
  if (path.basename(projectRoot).toLowerCase() === 'app' || hasAndroidAppModule(projectRoot)) {
    throw new Error(
      `Refusing to apply fixes inside an Android app module:\n${projectRoot}\n\n` +
      `This would create a duplicate app/app structure.\n` +
      `Move apply-fixes.js to the real project root and run it from there.`
    );
  }

  if (!hasAndroidProjectRoot(projectRoot)) {
    throw new Error(
      `Invalid target root:\n${projectRoot}\n\n` +
      `Expected:\n` +
      `  settings.gradle.kts\n` +
      `  app/src/main/AndroidManifest.xml`
    );
  }
}

function getDestinationBaseDir(extractedRoot) {
  const extractedIsFullProject =
    hasFile(extractedRoot, 'settings.gradle.kts') ||
    hasFile(extractedRoot, 'settings.gradle');

  const extractedIsAppOnly = hasAndroidAppModule(extractedRoot);

  if (extractedIsFullProject) {
    return PROJECT_ROOT;
  }

  if (extractedIsAppOnly) {
    return path.join(PROJECT_ROOT, 'app');
  }

  return PROJECT_ROOT;
}

function assertNoDuplicateAppDestination(sourceFiles, destinationBaseDir) {
  for (const file of sourceFiles) {
    const destFile = path.join(destinationBaseDir, file.relativePath);
    const normalized = path.normalize(destFile).toLowerCase();

    const badPath = path.normalize(path.join(PROJECT_ROOT, 'app', 'app')).toLowerCase();

    if (normalized.startsWith(badPath + path.sep) || normalized === badPath) {
      throw new Error(
        `Refusing to create duplicate app/app path.\n\n` +
        `Bad destination:\n${destFile}\n\n` +
        `This means the ZIP is being applied to the wrong folder.`
      );
    }
  }
}

async function applyFixes(zipFileName, options = {}) {
  if (!zipFileName) {
    throw new Error('Missing ZIP filename.');
  }

  assertSafeTargetRoot();

  const stagedZipPath = stageZipFromDownloads(zipFileName);

  cleanupTempDir();

  console.log(`\nExtracting ZIP: ${stagedZipPath}`);
  expandZip(stagedZipPath, TEMP_EXTRACT_DIR);

  const extractedProjectRoot = findExtractedProjectRoot(TEMP_EXTRACT_DIR);
  console.log(`Found extracted root: ${extractedProjectRoot}`);

  const destinationBaseDir = getDestinationBaseDir(extractedProjectRoot);
  console.log(`Destination base: ${destinationBaseDir}`);

  const sourceFiles = collectFiles(extractedProjectRoot);
  assertNoDuplicateAppDestination(sourceFiles, destinationBaseDir);

  const results = [];
  const timestamp = new Date().toISOString().replace(/[:.]/g, '-');
  const currentBackupDir = path.join(BACKUP_ROOT, timestamp);

  console.log(`\nApplying ${sourceFiles.length} files...`);

  for (const file of sourceFiles) {
    const sourceFile = file.fullPath;
    const destFile = path.join(destinationBaseDir, file.relativePath);
    const relativeToProject = path.relative(PROJECT_ROOT, destFile);

    const result = {
      action: 'skipped',
      file: relativeToProject,
      error: null
    };

    if (!exists(destFile)) {
      if (!options.createNew) {
        const createIt = await askYesNo(
          `\n[?] Missing file: ${relativeToProject}\nCreate it? (y/n): `
        );

        if (!createIt) {
          result.action = 'error';
          result.error = 'File does not exist and creation was declined.';
          results.push(result);
          continue;
        }
      }

      ensureDir(path.dirname(destFile));
      fs.copyFileSync(sourceFile, destFile);

      result.action = 'created';
      result.oldLines = 0;
      result.newLines = getLineCount(sourceFile);
      results.push(result);
      continue;
    }

    if (filesAreEqual(sourceFile, destFile)) {
      results.push(result);
      continue;
    }

    const oldLines = getLineCount(destFile);
    const newLines = getLineCount(sourceFile);

    const backupFile = path.join(currentBackupDir, relativeToProject);
    ensureDir(path.dirname(backupFile));
    fs.copyFileSync(destFile, backupFile);

    ensureDir(path.dirname(destFile));
    fs.copyFileSync(sourceFile, destFile);

    result.action = 'updated';
    result.oldLines = oldLines;
    result.newLines = newLines;
    results.push(result);
  }

  cleanupTempDir();
  printResults(results, currentBackupDir);
}

function revertFixes(timestamp) {
  assertSafeTargetRoot();

  const targetBackupDir = path.join(BACKUP_ROOT, timestamp);

  if (!exists(targetBackupDir)) {
    console.error(`\nAvailable backups in ${BACKUP_ROOT}:`);

    if (exists(BACKUP_ROOT)) {
      fs.readdirSync(BACKUP_ROOT).forEach(dir => {
        console.log(` - ${dir}`);
      });
    } else {
      console.log('  No backups found.');
    }

    throw new Error(`Backup folder not found: ${targetBackupDir}`);
  }

  console.log(`\nReverting files from backup: ${timestamp}`);

  const backupFiles = collectFiles(targetBackupDir);
  const results = [];

  for (const file of backupFiles) {
    const backupFile = file.fullPath;
    const destFile = path.join(PROJECT_ROOT, file.relativePath);
    const relativeToProject = path.relative(PROJECT_ROOT, destFile);

    ensureDir(path.dirname(destFile));
    fs.copyFileSync(backupFile, destFile);

    results.push({
      action: 'reverted',
      file: relativeToProject
    });
  }

  for (const res of results) {
    console.log(`REVERTED  ${res.file}`);
  }

  console.log(`\nSuccessfully reverted ${results.length} files.`);
}

function printResults(results, backupDir) {
  const updatedCount = results.filter(r => r.action === 'updated').length;
  const createdCount = results.filter(r => r.action === 'created').length;
  const skippedCount = results.filter(r => r.action === 'skipped').length;
  const errorResults = results.filter(r => r.action === 'error');

  console.log('\n--- Operations Log ---');

  for (const res of results) {
    if (res.action === 'updated') {
      let warning = '';

      if (res.oldLines > 0) {
        const decrease = res.oldLines - res.newLines;
        const percentDecrease = decrease / res.oldLines;

        if (percentDecrease > 0.1) {
          warning = '  WARNING: line count reduced by more than 10%';
        }
      }

      console.log(
        `UPDATED   ${res.file} ` +
        `(Lines: ${res.oldLines} -> ${res.newLines})${warning}`
      );
    } else if (res.action === 'created') {
      console.log(`CREATED   ${res.file} (Lines: ${res.newLines})`);
    } else if (res.action === 'skipped') {
      console.log(`SKIPPED   ${res.file} (no changes)`);
    } else if (res.action === 'error') {
      console.error(`ERROR     ${res.file} -> ${res.error}`);
    }
  }

  console.log('\n--- Summary ---');
  console.log(`Updated: ${updatedCount}`);
  console.log(`Created: ${createdCount}`);
  console.log(`Skipped: ${skippedCount}`);
  console.log(`Errors:  ${errorResults.length}`);

  if (updatedCount > 0 || createdCount > 0) {
    console.log(`\nBackup saved.`);
    console.log(`To revert, run:`);
    console.log(`node apply-fixes.js revert ${path.basename(backupDir)}`);
  } else {
    console.log('\nNo files were changed.');

    if (exists(backupDir) && fs.readdirSync(backupDir).length === 0) {
      fs.rmSync(backupDir, {
        recursive: true,
        force: true
      });
    }
  }

  if (errorResults.length > 0) {
    process.exitCode = 1;
  }
}

async function main() {
  const command = process.argv[2];
  const target = process.argv[3];
  const options = parseApplyOptions(process.argv.slice(4));

  if (command === 'apply' && target) {
    await applyFixes(target, options);
    return;
  }

  if (command === 'revert' && target) {
    revertFixes(target);
    return;
  }

  console.error('Usage:');
  console.error('  Apply fix:   node apply-fixes.js apply <zip-file-name-in-downloads>');
  console.error('  Revert fix:  node apply-fixes.js revert <timestamp-folder-name>');
  console.error('');
  console.error('Examples:');
  console.error('  node apply-fixes.js apply AIIPTVViewer-Trakt-UX-Full.zip');
  console.error('  node apply-fixes.js apply AIIPTVViewer-Trakt-UX-Full.zip --no-create-new');
  console.error('');
  console.error('Expected ZIP location:');
  console.error(`  ${DOWNLOADS_DIR}`);
  process.exit(1);
}

async function run() {
  try {
    await main();
  } catch (error) {
    console.error('\nERROR:', error.message);
    process.exitCode = 1;
  } finally {
    cleanupTempDir();
  }
}

run();

function getLocalIpAddresses(): string[] {
  const nets = networkInterfaces();
  const addresses: string[] = [];
  for (const name of Object.keys(nets)) {
    for (const net of nets[name] ?? []) {
      // Skip internal (127.0.0.1) and non-IPv4 addresses
      const isIPv4 = net.family === "IPv4" || (net.family as unknown) === 4;
      if (isIPv4 && !net.internal) {
        addresses.push(net.address);
      }
    }
  }
  return addresses;
}

// ===== server/src/middleware/rateLimit.ts =====
import type { NextFunction, Request, Response } from "express";

interface Bucket {
  count: number;
  resetAt: number;
}

export function rateLimit(options: { windowMs: number; max: number; keyPrefix?: string }) {
  const buckets = new Map<string, Bucket>();
  return (req: Request, res: Response, next: NextFunction): void => {
    const now = Date.now();
    if (buckets.size > 10_000) {
      for (const [key, bucket] of buckets) if (bucket.resetAt <= now) buckets.delete(key);
    }
    const key = `${options.keyPrefix ?? "default"}:${req.ip || req.socket.remoteAddress || "unknown"}`;
    const current = buckets.get(key);
    const bucket = !current || current.resetAt <= now ? { count: 0, resetAt: now + options.windowMs } : current;
    bucket.count += 1;
    buckets.set(key, bucket);
    res.setHeader("RateLimit-Limit", String(options.max));
    res.setHeader("RateLimit-Remaining", String(Math.max(0, options.max - bucket.count)));
    res.setHeader("RateLimit-Reset", String(Math.ceil(bucket.resetAt / 1000)));
    if (bucket.count > options.max) {
      res.setHeader("Retry-After", String(Math.max(1, Math.ceil((bucket.resetAt - now) / 1000))));
      res.status(429).json({ error: "Too many requests. Try again later." });
      return;
    }
    next();
  };
}

import { randomUUID } from "node:crypto";
import type { NextFunction, Request, Response } from "express";

export function requestContext(req: Request, res: Response, next: NextFunction): void {
  const requestId = req.header("x-request-id")?.slice(0, 100) || randomUUID();
  res.locals.requestId = requestId;
  res.setHeader("X-Request-Id", requestId);
  const startedAt = Date.now();
  res.on("finish", () => {
    console.info(JSON.stringify({
      level: "info",
      requestId,
      method: req.method,
      path: req.originalUrl,
      status: res.statusCode,
      durationMs: Date.now() - startedAt
    }));
  });
  next();
}

export function scopeProfileIds(req: Request, res: Response, next: NextFunction): void {
  const authUserId = res.locals.authUser?.id as string | undefined;
  if (!authUserId) {
    next();
    return;
  }
  const scope = (value: unknown): string | undefined => {
    if (typeof value !== "string" || !value.trim()) return undefined;
    const raw = value.trim();
    if (raw.startsWith(`${authUserId}:`)) return raw;
    const localId = raw.includes(":") ? raw.substring(raw.lastIndexOf(":") + 1) : raw;
    return `${authUserId}:${localId}`;
  };
  const queryProfile = scope(req.query.profileId ?? req.query.userId);
  if (queryProfile) {
    req.query.profileId = queryProfile;
    if (req.query.userId != null) req.query.userId = authUserId;
  }
  if (req.body && typeof req.body === "object") {
    const body = req.body as Record<string, unknown>;
    const bodyProfile = scope(body.profileId ?? body.userId);
    if (bodyProfile) body.profileId = bodyProfile;
    if (body.userId != null) body.userId = authUserId;
  }
  next();
}

function send(socket: WebSocket, payload: unknown): void {
  if (socket.readyState === socket.OPEN) {

  }

function safeString(value: unknown, maxLength: number): string | undefined {
  if (typeof value !== "string") return undefined;
  const trimmed = value.trim();
  return trimmed ? trimmed.slice(0, maxLength) : undefined;
}

function safeClientMessageId(value: unknown): string | undefined {
  return safeString(value, 128);
}

export function broadcastToRoom(roomId: string, payload: unknown, exceptSocket?: WebSocket): void {
  const sockets = rooms.get(roomId) ?? new Set<WebSocket>();
  for (const socket of sockets) {
    if (socket !== exceptSocket) send(socket, payload);
  }
}

export function broadcastSocialEvent(roomId: string, payload: unknown): void {
  broadcastToRoom(roomId, payload);
}

async function joinSocketRoom(client: ClientRecord, roomId: string): Promise<boolean> {
  if (client.rooms.has(roomId)) {
    await joinRoom(roomId, client.user.userId);
    return false;
  }

  client.rooms.add(roomId);
  if (!rooms.has(roomId)) rooms.set(roomId, new Set());
  rooms.get(roomId)!.add(client.socket);

  await joinRoom(roomId, client.user.userId);
  broadcastToRoom(roomId, { type: "userJoined", roomId, user: client.user }, client.socket);
  return true;
}

async function leaveSocketRoom(client: ClientRecord, roomId: string): Promise<void> {
  if (!client.rooms.has(roomId)) return;

  client.rooms.delete(roomId);
  rooms.get(roomId)?.delete(client.socket);
  await leaveRoom(roomId, client.user.userId);
  broadcastToRoom(roomId, { type: "userLeft", roomId, user: client.user }, client.socket);
}

function getToken(request: IncomingMessage): string | null {
  const host = request.headers.host ?? "localhost";
  const url = new URL(request.url ?? "/", `http://${host}`);
  const tokenFromQuery = url.searchParams.get("token");
  if (tokenFromQuery) return tokenFromQuery;

  const authHeader = request.headers.authorization ?? "";
  const [scheme, token] = authHeader.split(" ");
  if (scheme?.toLowerCase() === "bearer" && token) return token;
  return null;
}

function parseJsonEvent(raw: WebSocket.RawData): Record<string, unknown> | null {
  try {
    const parsed = JSON.parse(String(raw));
    return parsed && typeof parsed === "object" ? parsed as Record<string, unknown> : null;
  } catch {
    return null;
  }
}

export function attachSocialWebSocket(server: Server): void {
  const wss = new WebSocketServer({ noServer: true });

  server.on("upgrade", (request, socket, head) => {
    const pathname = new URL(request.url ?? "/", `http://${request.headers.host ?? "localhost"}`).pathname;
    if (pathname !== "/ws/social") return;

    const token = getToken(request);
    const payload = token ? verifySocialToken(token) : null;
    if (!payload) {
      socket.write("HTTP/1.1 401 Unauthorized\r\nConnection: close\r\n\r\n");
      socket.destroy();
      return;
    }

    wss.handleUpgrade(request, socket, head, (ws) => wss.emit("connection", ws, request, payload));
  });

  wss.on("connection", async (socket: WebSocket, _request: IncomingMessage, payload: ReturnType<typeof verifySocialToken>) => {
    if (!payload) {
      socket.close(1008, "Unauthorized");
      return;
    }

    const user = await ensureSocialUserByPayload({
      userId: payload.userId,
      username: payload.username,
      displayName: payload.displayName
    });
    const client: ClientRecord = { socket, user, rooms: new Set() };
    clients.set(socket, client);
    send(socket, { type: "connected", user, serverTime: new Date().toISOString() });

    socket.on("message", async (raw) => {
      const event = parseJsonEvent(raw);
      if (!event) {
        send(socket, { type: "error", error: "Invalid JSON event" });
        return;
      }

      try {
        const roomId = safeString(event.roomId, 256);

        if (event.type === "ping") {
          send(socket, { type: "pong", serverTime: new Date().toISOString() });
          return;
        }

        if (event.type === "joinRoom" && roomId) {
          await joinSocketRoom(client, roomId);
          send(socket, { type: "joinedRoom", roomId });
          return;
        }

        if (event.type === "leaveRoom" && roomId) {
          await leaveSocketRoom(client, roomId);
          send(socket, { type: "leftRoom", roomId });
          return;
        }

        if (event.type === "typing" && roomId) {
          if (!client.rooms.has(roomId)) await joinSocketRoom(client, roomId);
          broadcastToRoom(roomId, { type: "typing", roomId, user: client.user, at: new Date().toISOString() }, socket);
          return;
        }

        if (event.type === "watchPresence") {
          const presence = await upsertWatchPresence(client.user.userId, {
            catalogItemId: safeString(event.catalogItemId, 256),
            catalogKey: safeString(event.catalogKey, 256),
            catalogTitle: safeString(event.catalogTitle, 240),
            catalogType: safeString(event.catalogType, 64),
            roomId,
            positionMs: typeof event.positionMs === "number" || typeof event.positionMs === "string" ? event.positionMs : 0,
            durationMs: typeof event.durationMs === "number" || typeof event.durationMs === "string" ? event.durationMs : 0,
            playbackState: safeString(event.playbackState, 64),
            deviceName: safeString(event.deviceName, 120)
          });
          if (presence.roomId) {
            await joinSocketRoom(client, presence.roomId);
            broadcastToRoom(presence.roomId, { type: "watchPresence", roomId: presence.roomId, presence }, client.socket);
            send(socket, { type: "watchPresence", roomId: presence.roomId, presence });
          }
          return;
        }

        if (event.type === "stopWatching") {
          await clearWatchPresence(client.user.userId, safeString(event.catalogKey, 256));
          send(socket, { type: "stoppedWatching", catalogKey: safeString(event.catalogKey, 256) });
          return;
        }

        if (event.type === "getCurrentViewers") {
          const catalogKey = safeString(event.catalogItemId, 256) ?? safeString(event.catalogKey, 256);
          if (!catalogKey) {
            send(socket, { type: "error", error: "catalogItemId or catalogKey is required" });
            return;
          }
          send(socket, { type: "currentViewers", catalogKey, viewers: await getActiveViewersForCatalog(catalogKey) });
          return;
        }

        if (event.type === "sendMessage" && roomId) {
          await joinSocketRoom(client, roomId);
          const message = await createMessage(roomId, client.user.userId, {
            clientMessageId: safeClientMessageId(event.clientMessageId),
            text: typeof event.text === "string" ? event.text : "",
            messageType: safeString(event.messageType, 64),
            sharePayload: event.sharePayload
          });
          const responseEvent = { type: "messageCreated", roomId, message };
          broadcastToRoom(roomId, responseEvent, socket);
          send(socket, responseEvent);
          return;
        }

        send(socket, { type: "error", error: `Unknown or invalid event type: ${String(event.type ?? "")}` });
      } catch (error) {
        send(socket, { type: "error", error: error instanceof Error ? error.message : "Event failed" });
      }
    });

    socket.on("close", async () => {
      clients.delete(socket);
      for (const roomId of client.rooms) rooms.get(roomId)?.delete(socket);
      await touchSocialUser(client.user.userId, "offline");
    });

    socket.on("error", () => {
      clients.delete(socket);
    });
  });
}

function queryString(value: unknown): string | undefined {
  return typeof value === "string" && value.trim() ? value.trim() : undefined;
}

function queryNumber(value: unknown): number | undefined {
  const parsed = Number(value);
  return Number.isFinite(parsed) ? parsed : undefined;
}

function itemYear(item: { year?: string | null; title?: string | null; description?: string | null }): number | null {
  const haystack = [item.year, item.title, item.description].filter(Boolean).join(" ");
  const match = haystack.match(/\b(19\d{2}|20\d{2})\b/);
  if (!match) return null;
  const year = Number(match[1]);
  return Number.isFinite(year) ? year : null;
}

function itemGenres(item: { genreJson?: string | null; category?: string | null }): string[] {
  const genres = item.genreJson ? JSON.parse(item.genreJson) as string[] : [];
  return [...genres, item.category].filter((value): value is string => Boolean(value));
}

function cleanOptionLabel(value: string | undefined): string | undefined {
  return value?.replace(/\s+\(\d+\)$/, "").trim();
}

type PublicCatalogItem = ReturnType<typeof toPublicCatalogItem> & Record<string, unknown>;

function normalizeMatchTitle(value: unknown): string {
  return String(value ?? "")
    .toLowerCase()
    .replace(/\[[^\]]*]/g, " ")
    .replace(/\([^)]*\)/g, " ")
    .replace(/\b(4k|uhd|fhd|hd|sd|hdr|x264|x265|hevc|bluray|web[- ]?dl|multi|dubbed|subbed)\b/g, " ")
    .replace(/[^a-z0-9]+/g, " ")
    .trim();
}

function catalogTypeForTmdb(type: unknown): string {
  const value = String(type ?? "").toLowerCase();
  return value === "tv" || value === "series" ? "series" : "movie";
}

function findCatalogMatch(tmdbItem: Record<string, unknown>, catalogItems: PublicCatalogItem[]): PublicCatalogItem | undefined {
  const tmdbId = queryString(tmdbItem.tmdbId) ?? queryString(tmdbItem.metadataId);
  const type = catalogTypeForTmdb(tmdbItem.type ?? tmdbItem.contentType);
  const title = normalizeMatchTitle(tmdbItem.title);
  const year = queryString(tmdbItem.year);

  if (tmdbId) {
    const idMatch = catalogItems.find((item) =>
      item.type === type &&
      (String(item.tmdbId ?? "") === tmdbId ||
        (String(item.metadataProvider ?? "").toLowerCase().includes("tmdb") && String(item.metadataId ?? "") === tmdbId))
    );
    if (idMatch) return idMatch;
  }

  return catalogItems.find((item) => {
    if (item.type !== type) return false;
    const itemTitle = normalizeMatchTitle(item.title);
    if (!itemTitle || itemTitle !== title) return false;
    if (!year || !item.year) return true;
    return String(item.year) === year;
  }) ?? catalogItems.find((item) => {
    if (item.type !== type) return false;
    const itemTitle = normalizeMatchTitle(item.title);
    return title.length >= 4 && itemTitle.length >= 4 && (itemTitle.includes(title) || title.includes(itemTitle));
  });
}

function splitMatchedTmdbItems(tmdbItems: Record<string, unknown>[], catalogItems: PublicCatalogItem[]) {
  const matched: Record<string, unknown>[] = [];
  const unmatched: Record<string, unknown>[] = [];
  const seenMatched = new Set<string>();

  for (const tmdbItem of tmdbItems) {
    const match = findCatalogMatch(tmdbItem, catalogItems);
    if (match) {
      const key = String(match.id ?? match.stableKey ?? `${match.type}:${match.title}`);
      if (!seenMatched.has(key)) {
        seenMatched.add(key);
        matched.push({
          ...match,
          description: queryString(match.description) ?? queryString(tmdbItem.description) ?? "",
          coverUrl: queryString(match.coverUrl) ?? queryString(tmdbItem.coverUrl),
          backdropUrl: queryString(match.backdropUrl) ?? queryString(tmdbItem.backdropUrl),
          tmdbId: queryString(match.tmdbId) ?? queryString(tmdbItem.tmdbId),
          metadataProvider: queryString(match.metadataProvider) ?? "TMDb",
          metadataId: queryString(match.metadataId) ?? queryString(tmdbItem.metadataId),
          sourceLabel: `${queryString(tmdbItem.sourceLabel) ?? "TMDb"} - Matched to IPTV`
        });
      }
    } else {
      unmatched.push({
        ...tmdbItem,
        sourceLabel: `${queryString(tmdbItem.sourceLabel) ?? "TMDb"} - Unmatched`
      });
    }
  }

  return { matched, unmatched, items: [...matched, ...unmatched] };
}

catalogRoutes.get("/", async (req, res) => {
  const type = queryString(req.query.type);
  const category = cleanOptionLabel(queryString(req.query.category));
  const genre = cleanOptionLabel(queryString(req.query.genre));
  const search = queryString(req.query.search) ?? "";
  const profileId = queryString(req.query.profileId) ?? "default";
  const yearFrom = queryNumber(req.query.yearFrom);
  const yearTo = queryNumber(req.query.yearTo);
  const ratingFrom = queryNumber(req.query.ratingFrom);
  const effectiveRatingFrom = ratingFrom != null && ratingFrom > 0 ? ratingFrom : undefined;
  const status = queryString(req.query.status) ?? "all";
  const sort = queryString(req.query.sort) ?? "rating";
  const take = Math.min(Number(req.query.limit ?? 200) || 200, 500);
  const rawItems = await prisma.catalogItem.findMany({
    where: {
      isBlocked: false,
      ...(type ? { type } : {}),
      ...(category ? { category } : {}),
      ...(genre ? {
        OR: [
          { category: genre },
          { genreJson: { contains: `"${genre}"` } }
        ]
      } : {}),
      ...(effectiveRatingFrom != null ? { rating: { gte: effectiveRatingFrom } } : {}),
      ...(search ? {
        OR: [
          { title: { contains: search } },
          { description: { contains: search } },
          { category: { contains: search } },
          { year: { contains: search } }
        ]
      } : {})
    },
    select: safeCatalogSelect(),
    orderBy: [{ addedAt: "desc" }],
    take: 2000
  });

  const [history, progress, watchlist] = await Promise.all([
    prisma.watchHistory.findMany({ where: { profileId }, select: { catalogItemId: true, watchedAt: true, completed: true } }),
    prisma.watchProgress.findMany({ where: { profileId }, select: { catalogItemId: true, progress: true, completed: true, updatedAt: true } }),
    prisma.watchlistItem.findMany({ where: { profileId }, select: { catalogItemId: true, createdAt: true } })
  ]);
  const preferences = await prisma.userCatalogPreference.findMany({ where: { profileId }, select: { catalogItemId: true, rating: true, ignored: true } });
  const watchedIds = new Set(history.filter((item) => item.completed).map((item) => item.catalogItemId));
  const inProgressIds = new Set(progress.filter((item) => !item.completed && item.progress > 0).map((item) => item.catalogItemId));
  const watchlistIds = new Set(watchlist.map((item) => item.catalogItemId));
  type CatalogPreferenceSummary = { catalogItemId: string; rating: number | null; ignored: boolean };
  const preferenceById = new Map<string, CatalogPreferenceSummary>(
    (preferences as CatalogPreferenceSummary[]).map((item) => [item.catalogItemId, item])
  );
  const recentWatchById = new Map<string, number>();
  for (const item of history) recentWatchById.set(item.catalogItemId, Math.max(recentWatchById.get(item.catalogItemId) ?? 0, item.watchedAt.getTime()));
  for (const item of progress) recentWatchById.set(item.catalogItemId, Math.max(recentWatchById.get(item.catalogItemId) ?? 0, item.updatedAt.getTime()));

  const filtered = rawItems.filter((item) => {
    if (preferenceById.get(item.id)?.ignored) return false;
    const year = itemYear(item);
    if (yearFrom != null && year != null && year < yearFrom) return false;
    if (yearTo != null && year != null && year > yearTo) return false;
    if (genre && !itemGenres(item).some((value) => value.toLowerCase() === genre.toLowerCase())) return false;
    if (status === "watched" && !watchedIds.has(item.id)) return false;
    if (status === "unwatched" && (watchedIds.has(item.id) || inProgressIds.has(item.id))) return false;
    if (status === "inProgress" && !inProgressIds.has(item.id)) return false;
    if (status === "watchlist" && !watchlistIds.has(item.id)) return false;
    return true;
  });

  filtered.sort((left, right) => {
    if (sort === "title") return left.title.localeCompare(right.title);
    if (sort === "year") return (itemYear(right) ?? 0) - (itemYear(left) ?? 0);
    if (sort === "added") return right.addedAt.getTime() - left.addedAt.getTime();
    if (sort === "watched") return (recentWatchById.get(right.id) ?? 0) - (recentWatchById.get(left.id) ?? 0);
    if (sort === "watchlist") return Number(watchlistIds.has(right.id)) - Number(watchlistIds.has(left.id));
    return (right.rating ?? 0) - (left.rating ?? 0) || right.addedAt.getTime() - left.addedAt.getTime();
  });

  const items = filtered.slice(0, take).map((item) => {
    const publicItem = toPublicCatalogItem(item) as Record<string, unknown>;
    return {
      ...publicItem,
      watched: watchedIds.has(item.id),
      inProgress: inProgressIds.has(item.id),
      watchlisted: watchlistIds.has(item.id),
      userRating: preferenceById.get(item.id)?.rating ?? null,
      ignored: preferenceById.get(item.id)?.ignored ?? false
    };
  });
  res.json({ items, total: filtered.length, limit: take });
});

    update: {},
    create: { id: profileId, name: profileId }
  });

function asString(value: unknown): string | null { return typeof value === "string" && value.trim() ? value.trim() : null; }
function asDate(value: unknown): Date | null {
  if (typeof value === "number") return new Date(value > 9_999_999_999 ? value : value * 1000);
  if (typeof value === "string") {
    const numeric = Number(value);
    if (Number.isFinite(numeric)) return new Date(numeric > 9_999_999_999 ? numeric : numeric * 1000);
    const parsed = new Date(value);
    if (!Number.isNaN(parsed.getTime())) return parsed;
  }
  return null;
}

    ? req.body.stableKeys.map(String).map((key) => key.trim()).filter(Boolean).slice(0, 500)

  if (rating !== undefined && ![-1, 0, 1].includes(rating)) {
    res.status(400).json({ error: "rating must be -1, 0, or 1." });
    return;
  }
  if (rating === undefined && ignored === undefined) {
    res.status(400).json({ error: "rating or ignored is required." });
    return;
  }
  if (stableKeys.length === 0) {
    res.status(400).json({ error: "stableKeys is required." });
    return;
  }

  await upsertProfile(profileId, res.locals.authUser?.id);
  const items = await prisma.catalogItem.findMany({ where: { stableKey: { in: stableKeys } }, select: { id: true } });
  const saved = await Promise.all(items.map((item) =>
    prisma.userCatalogPreference.upsert({
      where: { profileId_catalogItemId: { profileId, catalogItemId: item.id } },
      update: {
        ...(rating !== undefined ? { rating: rating === 0 ? null : rating } : {}),
        ...(ignored !== undefined ? { ignored } : {})
      },
      create: {
        profileId,
        catalogItemId: item.id,
        rating: rating === undefined || rating === 0 ? null : rating,
        ignored: ignored ?? false
      }
    })
  ));

  res.json({ saved: saved.length });

pairingRoutes.post("/start", async (req, res) => {
  const deviceName = typeof req.body?.deviceName === "string" ? req.body.deviceName.trim().slice(0, 80) : null;
  const now = Date.now();
  const expiresAt = new Date(now + pairingMinutes * 60_000);

  await prisma.devicePairing.create({

      expiresAt
    }
  });

  res.status(201).json({

    qrPayload: `${baseUrl}/pair?code=${encodeURIComponent(deviceCode)}`,
    expiresAt: expiresAt.toISOString(),
    pollIntervalSeconds: 2
  });
});

pairingRoutes.get("/status", async (req, res) => {
  const deviceCode = normalizeCode(typeof req.query.deviceCode === "string" ? req.query.deviceCode : "");
  if (!deviceCode) {
    res.status(400).json({ error: "deviceCode is required." });
    return;
  }

  const pairing = await prisma.devicePairing.findUnique({ where: { deviceCode } });
  if (!pairing) {
    res.status(404).json({ status: "not_found" });
    return;
  }

  if (pairing.status === "pending" && pairing.expiresAt.getTime() <= Date.now()) {
    await prisma.devicePairing.update({
      where: { id: pairing.id },
      data: { status: "expired" }
    });
    res.json({ status: "expired" });
    return;
  }

  if (pairing.status === "approved" && pairing.deviceToken && pairing.username) {
    if (!pairing.consumedAt) {
      await prisma.devicePairing.update({
        where: { id: pairing.id },
        data: { consumedAt: new Date() }
      });
    }
    res.json({
      status: "approved",
      syncServerUrl: publicBaseUrl(req),
      syncServerUsername: pairing.username,
      syncServerToken: pairing.deviceToken,
      displayName: pairing.displayName,
      expiresAt: pairing.expiresAt.toISOString()
    });
    return;
  }

  res.json({
    status: pairing.status,
    expiresAt: pairing.expiresAt.toISOString()
  });
});

pairingRoutes.post("/approve", requireServerUser, async (req, res) => {
  const deviceCode = normalizeCode(typeof req.body?.deviceCode === "string" ? req.body.deviceCode : "");
  if (!deviceCode) {
    res.status(400).json({ error: "deviceCode is required." });
    return;
  }

  const pairing = await prisma.devicePairing.findUnique({ where: { deviceCode } });
  if (!pairing) {
    res.status(404).json({ error: "Pairing code was not found." });
    return;
  }
  if (pairing.status !== "pending") {
    res.status(409).json({ error: `Pairing code is already ${pairing.status}.` });
    return;
  }
  if (pairing.expiresAt.getTime() <= Date.now()) {
    await prisma.devicePairing.update({ where: { id: pairing.id }, data: { status: "expired" } });
    res.status(410).json({ error: "Pairing code expired. Start pairing again on the TV." });
    return;
  }

  const authUser = res.locals.authUser;
  if (!authUser) {
    res.status(401).json({ error: "Server login required to approve pairing." });
    return;
  }

  const updated = await prisma.devicePairing.update({
    where: { id: pairing.id },
    data: {
      status: "approved",
      serverUserId: authUser.id,
      username: authUser.username,
      displayName: authUser.displayName,
      deviceToken: randomBytes(32).toString("base64url"),
      approvedAt: new Date()
    }
  });

  res.json({
    status: updated.status,
    deviceName: updated.deviceName,
    username: updated.username,
    displayName: updated.displayName,
    approvedAt: updated.approvedAt?.toISOString()
  });
});

pairingRoutes.post("/cancel", async (req, res) => {
  const deviceCode = normalizeCode(typeof req.body?.deviceCode === "string" ? req.body.deviceCode : "");
  if (!deviceCode) {
    res.status(400).json({ error: "deviceCode is required." });
    return;
  }

  await prisma.devicePairing.updateMany({
    where: { deviceCode, status: "pending" },
    data: { status: "cancelled" }
  });
  res.json({ ok: true });
});

async function uniqueDeviceCode(): Promise<string> {
  for (let attempt = 0; attempt < 10; attempt += 1) {
    const code = generateCode();
    const existing = await prisma.devicePairing.findUnique({ where: { deviceCode: code } });
    if (!existing) return code;
  }
  throw new Error("Could not generate a pairing code.");
}

function generateCode(): string {
  const raw = randomBytes(6);
  const chars = Array.from(raw, (byte) => codeAlphabet[byte % codeAlphabet.length]);
  return `${chars.slice(0, 3).join("")}-${chars.slice(3, 6).join("")}`;
}

function normalizeCode(value: string): string {
  const cleaned = value.toUpperCase().replace(/[^A-Z0-9]/g, "");
  return cleaned.length === 6 ? `${cleaned.slice(0, 3)}-${cleaned.slice(3)}` : value.toUpperCase().trim();
}

function publicBaseUrl(req: Request): string {
  if (config.publicBaseUrl.trim()) return config.publicBaseUrl.trim().replace(/\/$/, "");
  const forwardedProto = req.header("x-forwarded-proto")?.split(",")[0]?.trim();
  const proto = forwardedProto || req.protocol || "http";
  return `${proto}://${req.get("host")}`;
}

    ? await prisma.viewingParty.update({ where: { id: party.id }, data: { playbackState, currentPositionMs: positionMs, durationMs, controlledByUserId: user.userId, lastControlAt: new Date() } })
    : await prisma.viewingParty.create({
        data: {
          roomId: room.id,
          createdByUserId: user.userId,
          catalogItemId: room.catalogItemId,
          catalogKey: room.catalogItemId ?? undefined,
          catalogTitle: room.catalogTitle ?? room.title,
          catalogType: room.catalogType,
          title: room.title,
          status: "active",
          playbackState,
          currentPositionMs: positionMs,
          durationMs,
          controlledByUserId: user.userId,
          lastControlAt: new Date()
        }
      });
  const payload = { type: "watchPartyControl", roomId: room.id, action, playbackState, positionMs, durationMs, controlledByUserId: user.userId, lastControlAt: updatedParty.lastControlAt };
  broadcastSocialEvent(room.id, payload);
  res.json({ party: updatedParty, control: payload });
});

  const username = typeof req.body?.username === "string" ? req.body.username.trim() : "";
  const password = typeof req.body?.password === "string" ? req.body.password : "";
  const displayName = typeof req.body?.displayName === "string" ? req.body.displayName.trim() : "";
  const isAdmin = req.body?.isAdmin === true;
  if (!username || username.length < 3) {
    res.status(400).json({ error: "Username must be at least 3 characters." });
    return;
  }
  if (!password || password.length < 8) {
    res.status(400).json({ error: "Password must be at least 8 characters." });
    return;
  }

  const user = await prisma.serverUser.create({
    data: {
      username,
      passwordHash: hashPassword(password),
      displayName: displayName || username,
      isAdmin
    },
    select: {
      id: true,
      username: true,
      displayName: true,
      isAdmin: true,
      createdAt: true
    }
  });
  res.status(201).json({ user });
});

  durationMinutes: number | null;
  traktSlug: string | null;
  traktType: string | null;
  imdbId: string | null;
  tmdbId: string | null;
  addedAt: Date;
};

    coverUrl: null;

function parseGenres(item: { genreJson: string | null; category: string | null }): string[] {
  const genres = item.genreJson ? JSON.parse(item.genreJson) as string[] : [];
  return [...genres, item.category].filter((value): value is string => Boolean(value));
}

function cleanTitle(title: string): string {
  return title
    .toLowerCase()
    .replace(/\[[^\]]*]/g, " ")
    .replace(/\([^)]*\)/g, " ")
    .replace(/\b(nf|netflix|amzn|amazon|hbo|max|disney|apple\s*tv|itunes)\b/gi, " ")
    .replace(/\b(uhd|4k|fhd|hd|sd|hdr|dv|h\.?264|h\.?265|x264|x265|hevc|bluray|blu ray|web[- ]?dl|webrip|brrip|hdrip|1080p|720p|2160p)\b/gi, " ")
    .replace(/\b(s\d{1,2}|season\s*\d{1,2}|complete|multi|dual audio|subbed|dubbed)\b/gi, " ")
    .replace(/\(\d{4}\)/g, "")
    .replace(/\[\d{4}\]/g, "")
    .replace(/\b(19|20)\d{2}\b/g, "")
    .replace(/[^a-z0-9]/g, "")
    .trim();
}

function titleTokenOverlap(left: string, right: string): number {
  const a = new Set(left.toLowerCase().split(/\W+/).filter((token) => token.length > 2));
  const b = new Set(right.toLowerCase().split(/\W+/).filter((token) => token.length > 2));
  if (!a.size || !b.size) return 0;
  return [...a].filter((token) => b.has(token)).length / Math.max(a.size, b.size);
}

async function resolveCatalogItem(title: string, type?: string, year?: string | null) {
  const cleanRecTitle = cleanTitle(title);
  if (!cleanRecTitle) return null;

  // 1. Try exact or contains matching at DB level
  const dbItems = await prisma.catalogItem.findMany({
    where: {
      isBlocked: false,
      type: type && type !== "any" ? type : undefined,
      title: { contains: title }
    },
    take: 10
  });

  const exact = dbItems.find((dbItem) => cleanTitle(dbItem.title) === cleanRecTitle);
  if (exact) return exact;
  const close = dbItems
    .map((dbItem) => ({ dbItem, overlap: titleTokenOverlap(dbItem.title, title) }))
    .filter((match) => match.overlap >= 0.72)
    .sort((left, right) => right.overlap - left.overlap)[0]?.dbItem;
  if (close) return close;

  // 2. Keyword fallback matching
  const keywords = title.split(/\s+/).filter(w => w.length > 3).slice(0, 3);
  if (keywords.length > 0) {
    const searchItems = await prisma.catalogItem.findMany({
      where: {
        isBlocked: false,
        type: type && type !== "any" ? type : undefined,
        OR: keywords.map(kw => ({ title: { contains: kw } }))
      },
      take: 20
    });
    const keywordExact = searchItems.find((dbItem) => cleanTitle(dbItem.title) === cleanRecTitle);
    if (keywordExact) return keywordExact;
    const keywordClose = searchItems
      .map((dbItem) => ({ dbItem, overlap: titleTokenOverlap(dbItem.title, title) }))
      .filter((match) => match.overlap >= 0.72)
      .sort((left, right) => right.overlap - left.overlap)[0]?.dbItem;
    if (keywordClose) return keywordClose;
  }

  return null;
}

function titleTokens(title: string): Set<string> {
  return new Set(title.toLowerCase().split(/\W+/).filter((token) => token.length > 3));
}

function tokenOverlapScore(candidateTitle: string, seedTitles: string[]): number {
  const candidateTokens = titleTokens(candidateTitle);
  if (candidateTokens.size === 0) return 0;
  let best = 0;
  for (const title of seedTitles) {
    const seedTokens = titleTokens(title);
    if (seedTokens.size === 0) continue;
    const overlap = [...candidateTokens].filter((token) => seedTokens.has(token)).length;
    best = Math.max(best, overlap / Math.max(candidateTokens.size, seedTokens.size));
  }
  return best;
}

function moodTerms(mood?: string): string[] {
  const terms: Record<string, string[]> = {
    funny: ["comedy", "sitcom", "funny"],
    action: ["action", "adventure", "thriller"],
    scary: ["horror", "scary", "supernatural"],
    chill: ["drama", "comfort", "light"],
    dateNight: ["romance", "comedy", "drama"],
    family: ["family", "animation", "kids"],
    binge: ["series", "drama", "crime"],
    short: ["short", "movie"],
    smart: ["sci-fi", "science", "mystery", "documentary"]
  };
  return terms[mood ?? "any"] ?? [];
}

function itemMatchesGenre(item: { genreJson: string | null; category: string | null }, genre?: string): boolean {
  if (!genre?.trim()) return true;
  const cleanRequested = genre.trim().toLowerCase().replace(/[^a-z0-9]/g, "");
  if (cleanRequested === "any") return true;

  return parseGenres(item).some((value) => {
    const cleanValue = value.toLowerCase().replace(/[^a-z0-9]/g, "");
    return cleanValue.includes(cleanRequested) || cleanRequested.includes(cleanValue);
  });
}

async function loadContext(profileId: string) {
  const inferredOwnerId = profileId.includes(":") ? profileId.slice(0, profileId.indexOf(":")) : undefined;
  const [profile, history, progress, watchlist, preferences, feedback] = await Promise.all([
    upsertProfile(profileId, inferredOwnerId),
    prisma.watchHistory.findMany({
      where: { profileId },
      include: { catalogItem: true },
      orderBy: { watchedAt: "desc" },
      take: 200
    }),
    prisma.watchProgress.findMany({
      where: { profileId },
      include: { catalogItem: true },
      orderBy: { updatedAt: "desc" },
      take: 100
    }),
    prisma.watchlistItem.findMany({
      where: { profileId },
      include: { catalogItem: true },
      orderBy: { createdAt: "desc" },
      take: 100
    }),
    prisma.userCatalogPreference.findMany({
      where: { profileId },
      include: { catalogItem: true },
      orderBy: { updatedAt: "desc" },
      take: 500
    }),
    prisma.recommendationFeedback.findMany({
      where: { profileId },
      include: { catalogItem: true },
      orderBy: { createdAt: "desc" },
      take: 500
    }).catch(() => [])
  ]);
  return { profile, history, progress, watchlist, preferences, feedback };
}

function deterministicScore(
  candidate: Candidate,
  context: Awaited<ReturnType<typeof loadContext>>,
  request: { mood?: string; contentType?: string; query?: string; mode?: string }
): ScoredCandidate {
  const completedItems = context.history.filter((entry) => entry.completed);
  const likedItems = context.preferences.filter((entry) => entry.rating === 1);
  const dislikedItems = context.preferences.filter((entry) => entry.rating === -1);
  const inProgressItems = context.progress.filter((entry) => !entry.completed && entry.progress > 0);
  const completedGenres = new Set(completedItems.flatMap((entry) => parseGenres(entry.catalogItem)));
  const likedGenres = new Set(likedItems.flatMap((entry) => parseGenres(entry.catalogItem)));
  const dislikedGenres = new Set(dislikedItems.flatMap((entry) => parseGenres(entry.catalogItem)));
  const watchlistGenres = new Set(context.watchlist.flatMap((entry) => parseGenres(entry.catalogItem)));
  const inProgressGenres = new Set(inProgressItems.flatMap((entry) => parseGenres(entry.catalogItem)));
  const candidateGenres = parseGenres(candidate);
  const likedAndWatchedTitles = [...likedItems, ...completedItems].map((entry) => entry.catalogItem.title);
  const watchlistTitles = context.watchlist.map((entry) => entry.catalogItem.title);
  const signals: string[] = [];
  let score = 0.25;

  for (const genre of candidateGenres) {
    if (completedGenres.has(genre)) {
      score += 0.18;
      signals.push(`watched ${genre}`);
    }
    if (likedGenres.has(genre)) {
      score += 0.28;
      signals.push(`liked ${genre}`);
    }
    if (dislikedGenres.has(genre)) {
      score -= 0.24;
      signals.push(`avoids disliked ${genre}`);
    }
    if (watchlistGenres.has(genre)) {
      score += 0.2;
      signals.push(`watchlist ${genre}`);
    }
    if (inProgressGenres.has(genre)) {
      score += 0.1;
      signals.push(`in-progress ${genre}`);
    }
  }

  const likedTitleOverlap = tokenOverlapScore(candidate.title, likedAndWatchedTitles);
  if (likedTitleOverlap >= 0.34) {
    score += Math.min(0.18, likedTitleOverlap * 0.2);
    signals.push("similar to watched or liked titles");
  }

  const watchlistTitleOverlap = tokenOverlapScore(candidate.title, watchlistTitles);
  if (watchlistTitleOverlap >= 0.34) {
    score += Math.min(0.16, watchlistTitleOverlap * 0.18);
    signals.push("similar to watchlist");
  }

  const feedbackForItem = context.feedback.filter((entry) => entry.catalogItemId === candidate.id);
  if (feedbackForItem.some((entry) => entry.action === "less_like_this" || entry.action === "hide_reason")) {
    score -= 0.35;
    signals.push("recent negative feedback");
  }
  if (feedbackForItem.some((entry) => entry.action === "more_like_this" || entry.action === "clicked")) {
    score += 0.18;
    signals.push("recent positive feedback");
  }

  if (request.mode === "short" && candidate.durationMinutes && candidate.durationMinutes <= 100) {
    score += 0.15;
    signals.push("short mode");
  }
  if (request.mode === "family" && candidateGenres.some((genre) => /family|animation|kids/i.test(genre))) {
    score += 0.16;
    signals.push("family mode");
  }
  if (request.mode === "new" && Date.now() - candidate.addedAt.getTime() < 1000 * 60 * 60 * 24 * 60) {
    score += 0.16;
    signals.push("new to library");
  }

  const terms = moodTerms(request.mood);
  const text = `${candidate.title} ${candidate.category ?? ""} ${candidate.description ?? ""}`.toLowerCase();
  if (terms.some((term) => text.includes(term))) {
    score += 0.12;
    signals.push(`${request.mood} mood`);
  }

  if (request.query) {
    for (const term of request.query.toLowerCase().split(/\W+/).filter((term) => term.length > 3)) {
      if (text.includes(term)) {
        score += 0.05;
        signals.push(`query ${term}`);
      }
    }
  }

  if (candidate.rating) score += Math.min(candidate.rating / 100, 0.1);
  if (Date.now() - candidate.addedAt.getTime() < 1000 * 60 * 60 * 24 * 45) {
    score += 0.05;
    signals.push("recently added");
  }

  return {
    catalogItem: candidate,
    score: Math.max(0.05, Math.min(score, 0.95)),
    reason: signals.length > 0 ? `Matched ${signals.slice(0, 3).join(", ")}.` : "Recommended from your local catalog.",
    matchedSignals: [...new Set(signals)].slice(0, 8)
  };
}

async function buildCandidatePool(
  context: Awaited<ReturnType<typeof loadContext>>,
  request: { mood?: string; contentType?: string; query?: string; genre?: string; mode?: string }
): Promise<ScoredCandidate[]> {
  const completedIds = new Set(context.history.filter((entry) => entry.completed).map((entry) => entry.catalogItemId));
  const ignoredIds = new Set(context.preferences.filter((entry) => entry.ignored).map((entry) => entry.catalogItemId));
  const explicitRewatch = /\brewatch|watch again|again\b/i.test(request.query ?? "");

  const whereClause: any = {
    isBlocked: false
  };

  if (request.contentType && request.contentType !== "any") {
    whereClause.type = request.contentType;
  }

  if (request.genre && request.genre.trim() && request.genre !== "any") {
    const cleanGenre = request.genre.trim().toLowerCase().replace(/[^a-z0-9]/g, "");
    if (cleanGenre === "scifi" || cleanGenre === "sciencefiction") {
      whereClause.OR = [
        { category: { contains: "sci" } },
        { category: { contains: "fiction" } },
        { category: { contains: "scifi" } },
        { genreJson: { contains: "sci" } },
        { genreJson: { contains: "fiction" } },
        { genreJson: { contains: "scifi" } }
      ];
    } else {
      whereClause.OR = [
        { category: { contains: request.genre.trim() } },
        { genreJson: { contains: request.genre.trim() } }
      ];
    }
  }

  const items = await prisma.catalogItem.findMany({
    where: whereClause,
    select: safeCatalogSelect(),
    orderBy: [{ rating: "desc" }, { addedAt: "desc" }],
    take: 500
  }) as Candidate[];

  return items
    .filter((item) => !ignoredIds.has(item.id))
    .filter((item) => explicitRewatch || !completedIds.has(item.id))
    .filter((item) => itemMatchesGenre(item, request.genre))
    .filter((item) => {
      if (request.mode === "short") return !item.durationMinutes || item.durationMinutes <= 110 || item.type === "episode";
      if (request.mode === "family") return !/adult|xxx|18\+|horror|violent|mature/i.test(`${item.title} ${item.category ?? ""} ${item.description ?? ""}`);
      return true;
    })
    .map((item) => deterministicScore(item, context, request))
    .sort((left, right) => right.score - left.score);
}

function toPromptCandidate(candidate: ScoredCandidate): PromptCandidate {
  return {
    catalogItemId: candidate.catalogItem.id,
    title: candidate.catalogItem.title,
    type: candidate.catalogItem.type,
    year: candidate.catalogItem.year,
    genre: parseGenres(candidate.catalogItem),
    category: candidate.catalogItem.category,
    rating: candidate.catalogItem.rating
  };
}

function localResponse(scoredCandidates: ScoredCandidate[], limit: number, source = "localFallback") {
  return {
    items: scoredCandidates.slice(0, limit).map((candidate) => ({
      catalogItem: toPublicCatalogItem(candidate.catalogItem),
      score: candidate.score,
      reason: candidate.reason,
      aiReason: candidate.reason,
      matchedSignals: candidate.matchedSignals,
      source
    })),
    summary: "I found recommendations from your local catalog and viewing signals.",
    webSearchUsed: false
  };
}

async function buildAiRecommendationRequestContext(request: AiRecommendationRequest) {
  const context = await loadContext(request.profileId);
  const scoredCandidates = await buildCandidatePool(context, request);
  const topCandidates = scoredCandidates.slice(0, config.aiMaxCandidates);
  const likedTitles = context.preferences.filter((entry) => entry.rating === 1).map((entry) => entry.catalogItem.title).slice(0, 30);
  const completedTitles = context.history.filter((entry) => entry.completed).map((entry) => entry.catalogItem.title).slice(0, 30);
  const watchHistoryTitles: string[] = context.history
    .map((entry) => String(entry.catalogItem.title ?? "").trim())
    .filter((title) => title.length > 0)
    .slice(0, 50);
  const inProgressTitles: string[] = context.progress
    .filter((entry) => !entry.completed && entry.progress > 0)
    .map((entry) => entry.catalogItem.title)
    .slice(0, 30);
  const thumbsDownTitles = context.preferences.filter((entry) => entry.rating === -1).map((entry) => entry.catalogItem.title).slice(0, 30);
  const ignoredTitles = context.preferences.filter((entry) => entry.ignored).map((entry) => entry.catalogItem.title).slice(0, 30);
  const watchlistTitles = context.watchlist.map((entry) => entry.catalogItem.title).slice(0, 30);

  let webSearchResults: string[] = [];
  const shouldSearchWeb = request.useWebSearch;

  if (shouldSearchWeb) {
    const searchQuery = request.query?.trim() || 
      [
        request.genre && request.genre !== "any" ? request.genre : "",
        request.mood && request.mood !== "any" ? request.mood : "",
        request.contentType && request.contentType !== "any" ? request.contentType : "movies and shows",
        likedTitles.length > 0 ? `similar to ${likedTitles.slice(0, 3).join(", ")}` : "",
        watchlistTitles.length > 0 ? `for fans of ${watchlistTitles.slice(0, 2).join(", ")}` : ""
      ].filter(Boolean).join(" ");
      
    if (searchQuery) {
      try {
        const results = await searchWeb(searchQuery, { maxResults: config.aiMaxWebResults });
        webSearchResults = results.map((r) => `${r.title}: ${r.snippet}${r.url ? ` (${r.url})` : ""}`);
      } catch (err) {
        console.error("Failed to fetch web search results for AI context:", err);
      }
    }
  }

  const prompt = buildRecommendationPrompt({
    mood: request.mood,
    genre: request.genre,
    contentType: request.contentType,
    query: request.query,
    useWebSearch: request.useWebSearch,
    likedTitles: [...new Set([...likedTitles, ...completedTitles])].slice(0, 30),
    watchHistoryTitles: [...new Set(watchHistoryTitles)].slice(0, 50),
    inProgressTitles: [...new Set(inProgressTitles)].slice(0, 30),
    dislikedTitles: thumbsDownTitles,
    ignoredTitles,
    watchlistTitles,
    candidates: topCandidates.map((candidate) => toPromptCandidate(candidate)),
    webSearchResults
  });

  return { context, scoredCandidates, topCandidates, prompt, webSearchUsed: webSearchResults.length > 0 };
}

export async function previewAiRecommendationPrompt(request: AiRecommendationRequest) {
  const { scoredCandidates, topCandidates, prompt, webSearchUsed } = await buildAiRecommendationRequestContext(request);
  return {
    prompt,
    candidateCount: scoredCandidates.length,
    promptCandidateCount: topCandidates.length,
    webSearchUsed,
    promptTokensEstimate: estimateTokens(prompt)
  };
}

export async function getAiRecommendations(request: AiRecommendationRequest) {
  const { scoredCandidates, topCandidates, prompt: generatedPrompt, webSearchUsed } = await buildAiRecommendationRequestContext(request);
  const prompt = request.prompt?.trim() || generatedPrompt;

  if (config.aiLogPrompts) {
    console.info("AI recommendation prompt", prompt);
  }

  let resultJson = "";
  let source = "ai";
  try {
    const json = await createJsonCompletion(
      [
        { role: "user", content: prompt }
      ],
      aiRecommendationSchemaDescription,
      { model: config.openRouterModel, timeoutMs: config.aiRecommendationTimeoutMs }
    );
    const parsed = aiModelResponseSchema.parse(json);
    resultJson = JSON.stringify(parsed);
    const run = await prisma.recommendationRun.create({
      data: {
        profileId: request.profileId,
        query: request.query,
        mood: request.mood,
        contentType: request.contentType,
        useWebSearch: webSearchUsed,
        model: config.openRouterModel,
        promptTokensEstimate: estimateTokens(prompt),
        resultJson
      }
    });

    const recommendations: RecommendationResult[] = [];
    for (const item of parsed.recommendations) {
      let dbItem: Awaited<ReturnType<typeof resolveCatalogItem>> = null;
      if (item.catalogItemId) {
        dbItem = await prisma.catalogItem.findFirst({
          where: { id: item.catalogItemId, isBlocked: false }
        });
      }

      if (!dbItem) {
        dbItem = await resolveCatalogItem(item.title, item.type, item.year);
      }

      if (dbItem) {
        recommendations.push({
          catalogItem: toPublicCatalogItem(dbItem),
          catalogItemId: dbItem.id,
          score: item.score,
          reason: item.reason,
          aiReason: item.reason,
          matchedSignals: item.matchedSignals,
          source
        });
      }
    }

    const finalRecommendations = recommendations.slice(0, request.limit);

    await Promise.all(finalRecommendations
      .filter((rec) => rec.source !== "ai_external")
      .map((recommendation) =>
        prisma.recommendation.create({
          data: {
            profileId: request.profileId,
            catalogItemId: recommendation.catalogItemId,
            score: recommendation.score,
            reason: recommendation.reason,
            source,
            aiReason: recommendation.aiReason,
            matchedSignalsJson: JSON.stringify(recommendation.matchedSignals),
            recommendationRunId: run.id
          }
        })
      )
    );

    return {
      items: finalRecommendations,
      summary: parsed.summary,
      webSearchUsed
    };
  } catch (err) {
    const message = err instanceof Error ? err.message : "Unknown AI recommendation error";
    console.warn(`AI recommendation model unavailable; using local fallback: ${message}`);
    source = "localFallback";
    const fallback = localResponse(scoredCandidates, request.limit, source);
    await prisma.recommendationRun.create({
      data: {
        profileId: request.profileId,
        query: request.query,
        mood: request.mood,
        contentType: request.contentType,
        useWebSearch: webSearchUsed,
        model: config.openRouterModel,
        promptTokensEstimate: estimateTokens(prompt),
        resultJson: JSON.stringify(fallback)
      }
    });
    return {
      ...fallback,
      webSearchUsed
    };
  }
}

export async function explainRecommendation(profileId: string, catalogItemId: string) {
  const context = await loadContext(profileId);
  const item = await prisma.catalogItem.findUnique({ where: { id: catalogItemId } });
  if (!item) throw new Error("Catalog item not found");
  const scored = deterministicScore(item as Candidate, context, {});
  return {
    reason: `You might like this because ${scored.reason.toLowerCase()}`,
    signals: scored.matchedSignals
  };
}

export async function getSimilarRecommendations(request: SimilarRequest) {
  const item = await prisma.catalogItem.findUnique({ where: { id: request.catalogItemId } });
  if (!item) throw new Error("Catalog item not found");
  return getAiRecommendations({
    profileId: request.profileId,
    mood: "any",
    contentType: item.type as "movie" | "series" | "episode" | "live",
    limit: request.limit,
    useWebSearch: request.useWebSearch,
    query: `Find local catalog items similar to ${item.title}`
  });
}

export async function recordRecommendationFeedback(feedback: RecommendationFeedback) {
  const catalogItem = await prisma.catalogItem.findFirst({
    where: {
      OR: [
        { id: feedback.catalogItemId },
        { stableKey: feedback.catalogItemId },
        { streamId: feedback.catalogItemId }
      ]
    },
    select: { id: true }
  });
  if (!catalogItem) throw new Error("Catalog item not found");
  await upsertProfile(feedback.profileId);
  const saved = await prisma.recommendationFeedback.create({
    data: {
      profileId: feedback.profileId,
      catalogItemId: catalogItem.id,
      action: feedback.action,
      mode: feedback.mode
    }
  });
  return { ok: true, id: saved.id };
}

export async function naturalSearch(request: NaturalSearchRequest) {
  const query = request.query.toLowerCase();
  const interpretedFilters = {
    contentType: query.includes("movie") ? "movie" as const : query.includes("show") || query.includes("series") ? "series" as const : "any" as const,
    mood: query.includes("funny") ? "funny" as const : query.includes("scary") ? "scary" as const : query.includes("action") ? "action" as const : "any" as const,
    maxDurationMinutes: query.includes("90-minute") || query.includes("90 minute") ? 100 : undefined,
    excludeCompleted: query.includes("not watched") || query.includes("unwatched")
  };
  const response = await getAiRecommendations({
    profileId: request.profileId,
    mood: interpretedFilters.mood,
    contentType: interpretedFilters.contentType,
    limit: request.limit,
    useWebSearch: request.useWebSearch,
    query: request.query
  });
  return {
    interpretedFilters,
    items: response.items
  };
}

import { pbkdf2Sync, randomBytes, timingSafeEqual } from "node:crypto";
import type { Request, Response, NextFunction } from "express";
import { config } from "../config.js";
import { prisma } from "../lib/prisma.js";

const iterations = 120_000;
const keyLength = 32;
const digest = "sha256";

export function hashPassword(password: string): string {
  const salt = randomBytes(16).toString("hex");
  const hash = pbkdf2Sync(password, salt, iterations, keyLength, digest).toString("hex");
  return `pbkdf2:${iterations}:${salt}:${hash}`;
}

export function verifyPassword(password: string, passwordHash: string): boolean {
  const [scheme, rawIterations, salt, expectedHash] = passwordHash.split(":");
  if (scheme !== "pbkdf2" || !rawIterations || !salt || !expectedHash) return false;
  const calculated = pbkdf2Sync(password, salt, Number(rawIterations), keyLength, digest);
  const expected = Buffer.from(expectedHash, "hex");
  return expected.length === calculated.length && timingSafeEqual(expected, calculated);
}

    },
    create: {
      username: config.serverUsername,
      passwordHash: hashPassword(config.serverPassword),
      displayName: config.serverUsername,
      isAdmin: true
    }
  });
}

export async function requireServerUser(req: Request, res: Response, next: NextFunction): Promise<void> {
  const authorization = req.header("authorization") ?? "";
  const bearer = parseBearerToken(authorization);
  if (bearer) {
    const pairing = await prisma.devicePairing.findFirst({
      where: {
        deviceToken: bearer,
        status: "approved",
        serverUserId: { not: null }
      },
      include: { serverUser: true }
    });
    if (pairing?.serverUser) {
      res.locals.authUser = {
        id: pairing.serverUser.id,
        username: pairing.serverUser.username,
        displayName: pairing.serverUser.displayName,
        isAdmin: pairing.serverUser.isAdmin
      };
      next();
      return;
    }
  }

  const credentials = parseBasicAuth(authorization);
  if (!credentials) {
    reject(res, "Server login required");
    return;
  }

  const user = await prisma.serverUser.findUnique({ where: { username: credentials.username } });
  if (!user || !verifyPassword(credentials.password, user.passwordHash)) {
    reject(res, "Invalid server username or password");
    return;
  }

  res.locals.authUser = {
    id: user.id,
    username: user.username,
    displayName: user.displayName,
    isAdmin: user.isAdmin
  };
  next();
}

export function requireAdmin(req: Request, res: Response, next: NextFunction): void {
  if (res.locals.authUser?.isAdmin) {
    next();
    return;
  }
  res.status(403).json({ error: "Admin access required" });
}

function parseBasicAuth(header: string): { username: string; password: string } | null {
  const [scheme, encoded] = header.split(" ");
  if (scheme?.toLowerCase() !== "basic" || !encoded) return null;
  const decoded = Buffer.from(encoded, "base64").toString("utf8");
  const separator = decoded.indexOf(":");
  if (separator < 0) return null;
  return {
    username: decoded.slice(0, separator),
    password: decoded.slice(separator + 1)
  };
}

function parseBearerToken(header: string): string | null {
  const [scheme, token] = header.split(" ");
  if (scheme?.toLowerCase() !== "bearer" || !token) return null;
  return token.trim() || null;
}

function reject(res: Response, message: string): void {
  res.setHeader("WWW-Authenticate", "Basic, Bearer");
  res.status(401).json({ error: message });
}

function safeJsonParse<T>(value: string | null | undefined, fallback: T): T {
  if (!value) return fallback;
  try { return JSON.parse(value) as T; } catch { return fallback; }
}

function normalizeTitle(value: string): string {
  return value
    .toLowerCase()
    .normalize("NFD")
    .replace(/[\u0300-\u036f]/g, "")
    .replace(/\[[^\]]*]/g, " ")
    .replace(/\([^)]*\)/g, " ")
    .replace(/\b(4k|uhd|fhd|hd|sd|hdr|dv|x264|x265|hevc|h264|h265|bluray|blu ray|web[- ]?dl|webrip|brrip|hdrip|1080p|720p|2160p|multi|dubbed|subbed|complete)\b/gi, " ")
    .replace(/\b(19|20)\d{2}\b/g, " ")
    .replace(/[^a-z0-9]+/g, " ")
    .trim();
}

export function computeDuplicateKey(item: { providerAccountId?: string | null; type: string; title: string; parentSeriesTitle?: string | null; year?: string | null; seasonNumber?: number | null; episodeNumber?: number | null }): string | null {
  const title = normalizeTitle(item.parentSeriesTitle || item.title);
  if (title.length < 4) return null;
  if (item.type === "episode") {
    return [item.providerAccountId ?? "", "episode", title, item.seasonNumber ?? 0, item.episodeNumber ?? 0].join(":");
  }
  return [item.providerAccountId ?? "", item.type, title, item.year ?? ""].join(":");
}

export function computeQuality(item: { title: string; type: string; year?: string | null; rating?: number | null; genreJson?: string | null; category?: string | null; coverUrl?: string | null; backdropUrl?: string | null; metadataProvider?: string | null; tmdbId?: string | null; imdbId?: string | null; streamId?: string | null; providerAccountId?: string | null; description?: string | null }) {
  let score = 35;
  const tags: string[] = [];
  const issues: string[] = [];
  const text = `${item.title} ${item.category ?? ""} ${item.description ?? ""}`.toLowerCase();
  const genres = safeJsonParse<string[]>(item.genreJson, []);

  if (item.streamId || item.type === "series") { score += 16; tags.push("playable"); } else { score -= 20; issues.push("missing stream id"); }
  if (item.coverUrl) score += 8; else issues.push("missing poster");
  if (item.backdropUrl) score += 5;
  if (item.year) score += 5; else issues.push("missing year");
  if (item.rating) score += Math.min(13, Math.max(0, item.rating * 1.3));
  if (genres.length) score += 6; else issues.push("missing genres");
  if (item.metadataProvider || item.tmdbId || item.imdbId) { score += 8; tags.push("metadata"); } else issues.push("unmatched metadata");
  if (/4k|uhd|2160/.test(text)) { score += 10; tags.push("4k"); }
  else if (/fhd|1080|\bhd\b/.test(text)) { score += 4; tags.push("hd"); }
  if (/cam|telesync|\bts\b/.test(text)) { score -= 22; issues.push("likely cam/ts"); }
  if (/adult|xxx|18\+/.test(text)) { score -= 30; issues.push("adult/clutter risk"); }
  if (/duplicate|backup|test/.test(text)) { score -= 16; issues.push("clutter keyword"); }

  const finalScore = Math.max(0, Math.min(100, Math.round(score)));
  return {
    score: finalScore,
    grade: finalScore >= 85 ? "A" : finalScore >= 72 ? "B" : finalScore >= 58 ? "C" : finalScore >= 42 ? "D" : "F",
    tags: [...new Set(tags)],
    issues: [...new Set(issues)],
    duplicateKey: computeDuplicateKey(item)
  };
}

export async function refreshCatalogQuality(limit = 5000) {
  const items = await prisma.catalogItem.findMany({ where: { isBlocked: false }, take: limit, orderBy: { updatedAt: "desc" } });
  let updated = 0;
  for (const item of items) {
    const quality = computeQuality(item);
    await prisma.catalogItem.update({
      where: { id: item.id },
      data: {
        qualityScore: quality.score,
        qualityTagsJson: JSON.stringify({ grade: quality.grade, tags: quality.tags, issues: quality.issues }),
        duplicateKey: quality.duplicateKey
      }
    });
    updated += 1;
  }
  return { updated };
}

export async function getQualityReport() {
  await refreshCatalogQuality(5000);
  const [summary, lowQuality] = await Promise.all([
    prisma.catalogItem.groupBy({ by: ["type"], where: { isBlocked: false }, _count: { _all: true }, _avg: { qualityScore: true } }),
    prisma.catalogItem.findMany({ where: { isBlocked: false, qualityScore: { lt: 55 } }, select: safeCatalogSelect(), orderBy: [{ qualityScore: "asc" }, { title: "asc" }], take: 80 })
  ]);
  return {
    summary: summary.map((row) => ({ type: row.type, count: row._count._all, averageQuality: Math.round(row._avg.qualityScore ?? 0) })),
    lowQuality: lowQuality.map(toPublicCatalogItem)
  };
}

export async function getDuplicateGroups(take = 80) {
  await refreshCatalogQuality(5000);
  const items = await prisma.catalogItem.findMany({
    where: { isBlocked: false, duplicateKey: { not: null } },
    select: { ...safeCatalogSelect(), qualityScore: true, qualityTagsJson: true, duplicateKey: true },
    orderBy: [{ duplicateKey: "asc" }, { qualityScore: "desc" }],
    take: 5000
  });
  const groups = new Map<string, typeof items>();
  for (const item of items) {
    if (!item.duplicateKey) continue;
    const group = groups.get(item.duplicateKey) ?? [];
    group.push(item);
    groups.set(item.duplicateKey, group);
  }
  return [...groups.values()]
    .filter((group) => group.length > 1)
    .map((group) => {
      const sorted = group.sort((a, b) => (b.qualityScore ?? 0) - (a.qualityScore ?? 0));
      return { key: sorted[0].duplicateKey, title: sorted[0].title, count: sorted.length, best: toPublicCatalogItem(sorted[0]), items: sorted.map(toPublicCatalogItem) };
    })
    .sort((a, b) => b.count - a.count || a.title.localeCompare(b.title))
    .slice(0, take);
}

export async function hideLowerQualityDuplicates() {
  const groups = await getDuplicateGroups(500);
  let hidden = 0;
  for (const group of groups) {
    const bestId = String((group.best as { id: string }).id);
    const ids = group.items.map((item) => String((item as { id: string }).id)).filter((id) => id !== bestId);
    if (ids.length) {
      await prisma.catalogItem.updateMany({ where: { id: { in: ids } }, data: { isBlocked: true, hiddenDuplicateOfId: bestId } });
      hidden += ids.length;
    }
  }
  return { hidden, groups: groups.length };
}

function asString(value: unknown): string | null {

export async function ingestWatchSnapshot(snapshot: Record<string, unknown>) {
  const eventId = asString(snapshot.eventId);
  const profileId = asString(snapshot.profileId) ?? "default";
  const userId = asString(snapshot.userId);
  await upsertProfile(profileId, userId);
  if (!await reserveSyncEvent(eventId, userId, profileId, "watch_snapshot")) return;

  try {
  const watchlist = Array.isArray(snapshot.watchlist) ? snapshot.watchlist as IncomingItem[] : [];
  const watchNext = Array.isArray(snapshot.watchNext) ? snapshot.watchNext as IncomingItem[] : [];
  const progress = Array.isArray(snapshot.progress) ? snapshot.progress as IncomingItem[] : [];
  const history = Array.isArray(snapshot.history) ? snapshot.history as IncomingItem[] : [];
  const finished = Array.isArray(snapshot.finishEpisodes) ? snapshot.finishEpisodes as IncomingItem[] : [];
  const catalog = Array.isArray(snapshot.catalog) ? snapshot.catalog as IncomingItem[] : [];

  const watchlistKeys = new Set([...watchlist, ...watchNext].map(itemStableKey));
  const progressKeys = new Set(progress.map(itemStableKey));
  const historyKeys = new Set([...history, ...finished].map(itemStableKey));

  for (const item of [...watchlist, ...watchNext, ...progress, ...history, ...finished, ...catalog]) {
    const catalogItem = await upsertCatalogItem(item);
    const key = itemStableKey(item);
    if (watchlistKeys.has(key)) {
      await prisma.watchlistItem.upsert({
        where: { profileId_catalogItemId: { profileId, catalogItemId: catalogItem.id } },
        update: {},
        create: { profileId, catalogItemId: catalogItem.id }
      });
    }
    if (progressKeys.has(key)) {
      const durationMs = asNumber(item.durationMs) ?? 0;
      const positionMs = asNumber(item.positionMs) ?? 0;
      const progressValue = durationMs > 0 ? Math.min(1, positionMs / durationMs) : (asNumber(item.progress) ?? 0);
      await prisma.watchProgress.upsert({
        where: { profileId_catalogItemId: { profileId, catalogItemId: catalogItem.id } },
        update: {
          progress: progressValue,
          positionMs,
          durationMs,
          completed: asBoolean(item.completed)
        },
        create: {
          profileId,
          catalogItemId: catalogItem.id,
          progress: progressValue,
          positionMs,
          durationMs,
          completed: asBoolean(item.completed)
        }
      });
    }
    if (historyKeys.has(key)) {
      await prisma.watchHistory.create({
        data: {
          profileId,
          catalogItemId: catalogItem.id,
          progress: asBoolean(item.completed) ? 1 : (asNumber(item.progress) ?? 0),
          completed: asBoolean(item.completed) || historyKeys.has(key),
          playCount: asNumber(item.watchedCount) ?? 1
        }
      });
    }
  }
  } catch (error) {
    await releaseSyncEvent(eventId);
    throw error;
  }
}

export async function ingestWatchEvent(event: Record<string, unknown>) {
  const eventId = asString(event.eventId);
  const profileId = asString(event.profileId) ?? "default";
  const userId = asString(event.userId);
  await upsertProfile(profileId, userId);
  if (!await reserveSyncEvent(eventId, userId, profileId, asString(event.eventType) ?? "watch_event")) return;
  try {
  const item = event.item && typeof event.item === "object" ? event.item as IncomingItem : event;
  const catalogItem = await upsertCatalogItem(item);
  const positionMs = asNumber(event.positionMs) ?? asNumber(item.positionMs) ?? 0;
  const durationMs = asNumber(event.durationMs) ?? asNumber(item.durationMs) ?? 0;
  const progress = asNumber(event.progress) ?? asNumber(item.progress) ?? (durationMs > 0 ? Math.min(1, positionMs / durationMs) : 0);
  const completed = asBoolean(event.completed) || progress >= 0.9;
  await prisma.watchHistory.create({
    data: {
      profileId,
      catalogItemId: catalogItem.id,
      progress,
      completed,
      playCount: 1
    }
  });
  await prisma.watchProgress.upsert({
    where: { profileId_catalogItemId: { profileId, catalogItemId: catalogItem.id } },
    update: { progress, positionMs, durationMs, completed },
    create: { profileId, catalogItemId: catalogItem.id, progress, positionMs, durationMs, completed }
  });
  } catch (error) {
    await releaseSyncEvent(eventId);
    throw error;
  }
}

  subject: string;
  text: string;
};

export async function sendMail(options: MailOptions): Promise<boolean> {
  if (!config.gmailUser || !config.gmailAppPassword || !looksLikeEmail(options.to)) {
    return false;
  }

  try {
    const nodemailer = require("nodemailer");
    const transporter = nodemailer.createTransport({
      service: "gmail",
      auth: {
        user: config.gmailUser,
        pass: config.gmailAppPassword
      }
    });

    await transporter.sendMail({
      from: config.gmailUser,
      ...options
    });
    return true;
  } catch (error) {
    const message = error instanceof Error ? error.message : "Unknown mail error";
    console.warn("Registration email was not sent:", message);
    return false;
  }
}

function looksLikeEmail(value: string): boolean {
  return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(value);
}

function normalizeTitle(value: string): string {
  return value
    .toLowerCase()
    .replace(/\[[^\]]*]/g, " ")
    .replace(/\([^)]*\)/g, " ")
    .replace(/\b(4k|uhd|fhd|hd|sd|hdr|x264|x265|hevc|bluray|web[- ]?dl)\b/gi, " ")
    .replace(/[^a-z0-9]+/g, " ")
    .trim();
}

function yearFrom(value?: string | null): string | null {
  const match = String(value ?? "").match(/\b(19\d{2}|20\d{2})\b/);
  return match?.[1] ?? null;
}

function imageUrl(path?: string | null, size = "w500"): string | null {
  return path ? `https://image.tmdb.org/t/p/${size}${path}` : null;
}

async function tmdbFetch(path: string, params: Record<string, string | number | undefined> = {}) {
  if (!config.tmdbApiKey) return null;
  const url = new URL(`https://api.themoviedb.org/3${path}`);
  for (const [key, value] of Object.entries(params)) {
    if (value != null && String(value).trim()) url.searchParams.set(key, String(value));
  }
  const headers: Record<string, string> = { accept: "application/json" };
  if (config.tmdbApiKey.startsWith("ey")) {
    headers.authorization = `Bearer ${config.tmdbApiKey}`;
  } else {
    url.searchParams.set("api_key", config.tmdbApiKey);
  }
  const response = await fetch(url, { headers });
  if (!response.ok) return null;
  return response.json() as Promise<Record<string, unknown>>;
}

async function tvmazeFetch(path: string, params: Record<string, string | number | undefined> = {}) {
  const url = new URL(`https://api.tvmaze.com${path}`);
  for (const [key, value] of Object.entries(params)) {
    if (value != null && String(value).trim()) url.searchParams.set(key, String(value));
  }
  const response = await fetch(url, { headers: { accept: "application/json" } });
  if (!response.ok) return null;
  return response.json() as Promise<unknown>;
}

function chooseTmdbResult(results: unknown, title: string, year?: string | null) {
  if (!Array.isArray(results)) return null;
  const normalized = normalizeTitle(title);
  const targetYear = yearFrom(year ?? title);
  return results.find((item) => {
    if (!item || typeof item !== "object") return false;
    const record = item as Record<string, unknown>;
    const itemTitle = String(record.title ?? record.name ?? "");
    const itemYear = yearFrom(String(record.release_date ?? record.first_air_date ?? ""));
    return normalizeTitle(itemTitle) === normalized && (!targetYear || itemYear === targetYear);
  }) ?? results.find((item) => {
    if (!item || typeof item !== "object") return false;
    const record = item as Record<string, unknown>;
    return normalizeTitle(String(record.title ?? record.name ?? "")).includes(normalized);
  }) ?? results[0] ?? null;
}

function tmdbGenres(value: unknown): string[] {
  if (!Array.isArray(value)) return [];
  return value
    .map((genre) => genre && typeof genre === "object" ? String((genre as Record<string, unknown>).name ?? "") : "")
    .filter(Boolean);
}

async function tmdbEnrichment(item: NonNullable<CatalogRecord>): Promise<Enrichment | null> {
  const mediaType = item.type === "series" || item.type === "episode" ? "tv" : "movie";
  const query = item.parentSeriesTitle || item.title;
  const search = await tmdbFetch(`/search/${mediaType}`, {
    query,
    year: mediaType === "movie" ? yearFrom(item.year ?? item.title) ?? undefined : undefined,
    first_air_date_year: mediaType === "tv" ? yearFrom(item.year ?? item.title) ?? undefined : undefined
  });
  const match = chooseTmdbResult(search?.results, query, item.year ?? item.title) as Record<string, unknown> | null;
  const id = match?.id;
  if (!id) return null;
  const details = await tmdbFetch(`/${mediaType}/${id}`, { append_to_response: "external_ids,keywords,credits" });
  if (!details) return null;
  const externalIds = details.external_ids && typeof details.external_ids === "object" ? details.external_ids as Record<string, unknown> : {};
  const genres = tmdbGenres(details.genres);
  return {
    title: String(details.title ?? details.name ?? item.title),
    year: yearFrom(String(details.release_date ?? details.first_air_date ?? item.year ?? "")),
    genreJson: genres.length ? JSON.stringify(genres) : item.genreJson,
    rating: typeof details.vote_average === "number" ? details.vote_average : item.rating,
    description: String(details.overview ?? item.description ?? ""),
    durationMinutes: typeof details.runtime === "number"
      ? details.runtime
      : Array.isArray(details.episode_run_time) && typeof details.episode_run_time[0] === "number"
        ? details.episode_run_time[0]
        : item.durationMinutes,
    coverUrl: imageUrl(String(details.poster_path ?? "")) ?? item.coverUrl,
    backdropUrl: imageUrl(String(details.backdrop_path ?? ""), "w1280") ?? item.backdropUrl,
    metadataProvider: "TMDb",
    metadataId: String(id),
    tmdbId: String(id),
    imdbId: typeof externalIds.imdb_id === "string" ? externalIds.imdb_id : item.imdbId
  };
}

async function tvmazeEnrichment(item: NonNullable<CatalogRecord>): Promise<Enrichment | null> {
  if (item.type !== "series" && item.type !== "episode") return null;
  const query = item.parentSeriesTitle || item.title;
  const results = await tvmazeFetch("/search/shows", { q: query });
  if (!Array.isArray(results) || results.length === 0) return null;
  const normalized = normalizeTitle(query);
  const row = results.find((entry) => {
    const show = entry && typeof entry === "object" ? (entry as Record<string, unknown>).show as Record<string, unknown> | undefined : undefined;
    return show && normalizeTitle(String(show.name ?? "")) === normalized;
  }) ?? results[0];
  const show = row && typeof row === "object" ? (row as Record<string, unknown>).show as Record<string, unknown> | undefined : undefined;
  if (!show?.id) return null;
  const image = show.image && typeof show.image === "object" ? show.image as Record<string, unknown> : {};
  const externals = show.externals && typeof show.externals === "object" ? show.externals as Record<string, unknown> : {};
  const genres = Array.isArray(show.genres) ? show.genres.map(String).filter(Boolean) : [];
  return {
    title: String(show.name ?? item.title),
    year: yearFrom(String(show.premiered ?? item.year ?? "")),
    genreJson: genres.length ? JSON.stringify(genres) : item.genreJson,
    rating: show.rating && typeof show.rating === "object" && typeof (show.rating as Record<string, unknown>).average === "number"
      ? (show.rating as Record<string, number>).average
      : item.rating,
    description: String(show.summary ?? item.description ?? "").replace(/<[^>]+>/g, " ").replace(/\s+/g, " ").trim(),
    durationMinutes: typeof show.runtime === "number" ? show.runtime : item.durationMinutes,
    coverUrl: typeof image.original === "string" ? image.original : typeof image.medium === "string" ? image.medium : item.coverUrl,
    metadataProvider: item.metadataProvider || "TVmaze",
    metadataId: item.metadataId || String(show.id),
    imdbId: typeof externals.imdb === "string" ? externals.imdb : item.imdbId
  };
}

function mergeEnrichment(item: NonNullable<CatalogRecord>, primary: Enrichment | null, fallback: Enrichment | null): Enrichment {
  const merged = { ...(fallback ?? {}), ...(primary ?? {}) };
  return {

    year: merged.year ?? item.year,
    genreJson: merged.genreJson ?? item.genreJson,
    rating: merged.rating ?? item.rating,
    description: merged.description || item.description,
    durationMinutes: merged.durationMinutes ?? item.durationMinutes,
    coverUrl: merged.coverUrl || item.coverUrl,
    backdropUrl: merged.backdropUrl || item.backdropUrl,
    metadataProvider: merged.metadataProvider ?? item.metadataProvider,
    metadataId: merged.metadataId ?? item.metadataId,
    imdbId: merged.imdbId ?? item.imdbId,
    tmdbId: merged.tmdbId ?? item.tmdbId
  };
}

export async function enrichCatalogItemWithExternalMetadata(id: string) {
  if (!config.metadataEnrichmentEnabled) throw new Error("Metadata enrichment is disabled");
  const item = await prisma.catalogItem.findUnique({ where: { id } });
  if (!item) throw new Error("Catalog item not found");
  const [tmdb, tvmaze] = await Promise.all([
    tmdbEnrichment(item).catch(() => null),
    tvmazeEnrichment(item).catch(() => null)
  ]);
  const data = mergeEnrichment(item, tmdb, tvmaze);
  const updated = await prisma.catalogItem.update({
    where: { id },
    data,
    select: safeCatalogSelect()
  });
  return toPublicCatalogItem(updated);
}

import { config } from "../config.js";

export type ChatMessage = {
  role: "system" | "user" | "assistant";
  content: string;
};

type CompletionOptions = {
  model?: string;
  temperature?: number;
  maxTokens?: number;
  timeoutMs?: number;
  jsonMode?: boolean;
};

type OpenRouterChoice = {
  message?: {
    content?: OpenRouterMessageContent;
  };
};

type OpenRouterMessageContent = string | Array<{ type?: string; text?: string }>;

function isTransientStatus(status: number): boolean {
  return [429, 500, 502, 503, 504].includes(status);
}

async function sleep(ms: number): Promise<void> {
  await new Promise((resolve) => setTimeout(resolve, ms));
}

export async function createChatCompletion(
  messages: ChatMessage[],
  options: CompletionOptions = {}
): Promise<string> {
  if (!config.openRouterApiKey) {
    throw new Error("OPENROUTER_API_KEY is not configured");
  }

  const model = options.model ?? config.openRouterModel;
  const timeoutMs = options.timeoutMs ?? config.aiRecommendationTimeoutMs;
  const url = `${config.openRouterBaseUrl.replace(/\/$/, "")}/chat/completions`;
  const body = {
    model,
    messages,
    temperature: options.temperature ?? 0.2,
    max_tokens: options.maxTokens ?? (options.jsonMode ? 3200 : 1400),
    ...(options.jsonMode ? { response_format: { type: "json_object" } } : {})
  };

  let lastError = "OpenRouter request failed";
  for (let attempt = 0; attempt < 2; attempt += 1) {
    const controller = new AbortController();
    const timeout = setTimeout(() => controller.abort(), timeoutMs);
    try {
      const response = await fetch(url, {
        method: "POST",
        headers: {
          Authorization: `Bearer ${config.openRouterApiKey}`,
          "Content-Type": "application/json",
          "HTTP-Referer": config.openRouterSiteUrl,
          "X-Title": config.openRouterAppName
        },
        body: JSON.stringify(body),
        signal: controller.signal
      });

      if (!response.ok) {
        lastError = `OpenRouter returned HTTP ${response.status}`;
        if (attempt === 0 && isTransientStatus(response.status)) {
          await sleep(750);
          continue;
        }
        throw new Error(lastError);
      }

      const data = (await response.json()) as { choices?: OpenRouterChoice[] };
      const content = data.choices
        ?.map((choice) => normalizeMessageContent(choice.message?.content))
        .find((value) => value.trim().length > 0);
      if (!content) {
        throw new Error("OpenRouter returned an empty response");
      }
      return content;
    } catch (error) {
      const message = error instanceof Error && error.name === "AbortError"
        ? "OpenRouter request timed out"
        : error instanceof Error
          ? error.message
          : "OpenRouter request failed";
      lastError = message;
      if (attempt === 0) {
        await sleep(750);
        continue;
      }
    } finally {
      clearTimeout(timeout);
    }
  }

  throw new Error(lastError);
}

function normalizeMessageContent(content: OpenRouterMessageContent | undefined): string {
  if (typeof content === "string") return content;
  if (Array.isArray(content)) {
    return content
      .map((part) => typeof part.text === "string" ? part.text : "")
      .join("")
      .trim();
  }
  return "";
}

export async function createJsonCompletion(
  messages: ChatMessage[],
  schemaDescription: string,
  options: CompletionOptions = {}
): Promise<unknown> {
  const response = await createChatCompletion(
    [
      {
        role: "system",
        content: `Return strict JSON only. The JSON shape is: ${schemaDescription}`
      },
      ...messages
    ],
    { ...options, jsonMode: true }
  );

  const trimmed = extractJsonPayload(response);
  try {
    return JSON.parse(trimmed);
  } catch {
    throw new Error(`OpenRouter returned invalid JSON: ${response.trim().slice(0, 220)}`);
  }
}

function extractJsonPayload(response: string): string {
  const withoutFence = response
    .trim()
    .replace(/^```(?:json)?\s*/i, "")
    .replace(/```\s*$/i, "")
    .trim();

  if (withoutFence.startsWith("{") && withoutFence.endsWith("}")) return withoutFence;

  const start = withoutFence.indexOf("{");
  if (start < 0) return withoutFence;

  let depth = 0;
  let inString = false;
  let escaped = false;
  for (let index = start; index < withoutFence.length; index += 1) {
    const char = withoutFence[index];
    if (escaped) {
      escaped = false;
      continue;
    }
    if (char === "\\") {
      escaped = true;
      continue;
    }
    if (char === "\"") {
      inString = !inString;
      continue;
    }
    if (inString) continue;
    if (char === "{") depth += 1;
    if (char === "}") {
      depth -= 1;
      if (depth === 0) return withoutFence.slice(start, index + 1);
    }
  }

  return withoutFence;
}

// ===== server/src/services/secretCrypto.ts =====
import { createCipheriv, createDecipheriv, createHash, randomBytes } from "node:crypto";
import { config } from "../config.js";

const PREFIX = "enc:v1";

function encryptionKey(): Buffer {
  const source = config.credentialEncryptionKey || config.serverPassword;
  if (!source) throw new Error("CREDENTIAL_ENCRYPTION_KEY is required before storing IPTV credentials");
  return createHash("sha256").update(source, "utf8").digest();
}

export function encryptSecret(value: string): string {
  if (!value || value.startsWith(`${PREFIX}:`)) return value;
  const iv = randomBytes(12);
  const cipher = createCipheriv("aes-256-gcm", encryptionKey(), iv);
  const encrypted = Buffer.concat([cipher.update(value, "utf8"), cipher.final()]);
  const tag = cipher.getAuthTag();
  return [PREFIX, iv.toString("base64url"), tag.toString("base64url"), encrypted.toString("base64url")].join(":");
}

export function decryptSecret(value: string): string {
  if (!value.startsWith(`${PREFIX}:`)) return value;
  const [, , ivText, tagText, encryptedText] = value.split(":");
  if (!ivText || !tagText || !encryptedText) throw new Error("Encrypted secret is malformed");
  const decipher = createDecipheriv("aes-256-gcm", encryptionKey(), Buffer.from(ivText, "base64url"));
  decipher.setAuthTag(Buffer.from(tagText, "base64url"));
  return Buffer.concat([
    decipher.update(Buffer.from(encryptedText, "base64url")),
    decipher.final()
  ]).toString("utf8");
}

function isLikelyStreamUrl(value: string | null): boolean {
  if (!value) return false;
  const lower = value.toLowerCase();
  return (
    lower.includes('/live/') ||
    lower.includes('/movie/') ||
    lower.includes('/series/') ||
    lower.includes('username=') ||
    lower.includes('password=') ||
    lower.endsWith('.m3u8') ||
    lower.endsWith('.ts') ||
    lower.includes('player_api.php')
  );
}

export async function getOrCreateRoom(input: {
  title?: string | null;
  roomType?: string | null;
  catalogItemId?: string | null;
  createdByUserId: string;
  participantUserIds?: string[];
}) {
  const roomType = input.roomType ?? "discussion";
  const catalogItemId = await resolveCatalogItemId(input.catalogItemId ?? null);
  if (catalogItemId && ["watchParty", "discussion"].includes(roomType)) {

      await joinRoom(existing.id, input.createdByUserId);
      for (const userId of input.participantUserIds ?? []) await joinRoom(existing.id, userId);
      return existing;
    }
  }

  const catalogItem = catalogItemId ? await prisma.catalogItem.findUnique({ where: { id: catalogItemId } }) : null;

  if (!value) return null;

    where: {
      OR: [
        { id: value },
        { stableKey: value },
        { streamId: value }
      ]
    },
    select: safeCatalogSelect()
  });
}

function asInt(value: unknown, fallback = 0): number {
  const parsed = Number(value);
  if (!Number.isFinite(parsed)) return fallback;
  return Math.max(0, Math.floor(parsed));
}

export async function requestFriendship(userId: string, otherUserId: string) {
  if (userId === otherUserId) throw new Error("Cannot friend yourself");
  const other = await prisma.socialUser.findUnique({ where: { id: otherUserId } });
  if (!other) throw new Error("Social user not found");
  const inverse = await prisma.socialFriendship.findUnique({
    where: { requesterUserId_addresseeUserId: { requesterUserId: otherUserId, addresseeUserId: userId } },
    include: { requester: true, addressee: true }
  });
  if (inverse?.status === "pending") {
    return prisma.socialFriendship.update({
      where: { id: inverse.id },
      data: { status: "accepted", respondedAt: new Date() },
      include: { requester: true, addressee: true }
    });
  }
  if (inverse) return inverse;
  return prisma.socialFriendship.upsert({
    where: { requesterUserId_addresseeUserId: { requesterUserId: userId, addresseeUserId: otherUserId } },
    update: { status: "pending", respondedAt: null },
    create: { requesterUserId: userId, addresseeUserId: otherUserId, status: "pending" },
    include: { requester: true, addressee: true }
  });
}

export async function respondToFriendship(userId: string, friendshipId: string, status: string) {
  const normalized = status === "accepted" ? "accepted" : status === "blocked" ? "blocked" : "declined";
  const friendship = await prisma.socialFriendship.findUnique({ where: { id: friendshipId } });
  if (!friendship || ![friendship.requesterUserId, friendship.addresseeUserId].includes(userId)) {
    throw new Error("Friend request not found");
  }
  return prisma.socialFriendship.update({
    where: { id: friendshipId },
    data: { status: normalized, respondedAt: new Date() },
    include: { requester: true, addressee: true }
  });
}

export async function createMessage(roomId: string, senderUserId: string, input: SocialMessageInput) {
  await joinRoom(roomId, senderUserId);
  const text = (asString(input.text) ?? "").slice(0, 2000);
  const messageType = asString(input.messageType) ?? (input.sharePayload ? "mediaShare" : "text");
  const clientMessageId = asString(input.clientMessageId);

  if (!text && !input.sharePayload && messageType !== "system") {
    throw new Error("Message cannot be empty");
  }

  if (clientMessageId) {
    const existing = await prisma.chatMessage.findFirst({
      where: { roomId, clientMessageId, senderUserId },
      include: { sender: true }
    });
    if (existing) return formatMessage(existing);
  }

  const message = await prisma.chatMessage.create({
    data: {
      roomId,
      senderUserId,
      clientMessageId,
      messageType,
      text,
      sharePayloadJson: safeJson(sanitizeSharePayload(messageType, input.sharePayload))
    },
    include: { sender: true }
  });

  await prisma.chatRoom.update({
    where: { id: roomId },
    data: {
      lastMessagePreview: text.slice(0, 160) || (messageType === "listShare" ? "Shared a list" : messageType === "mediaShare" ? "Shared a title" : "New message"),
      lastMessageAt: message.createdAt,
      updatedAt: new Date()
    }
  });

  return formatMessage(message);
}

export async function createSystemMessage(roomId: string, text: string) {
  const room = await prisma.chatRoom.findUnique({ where: { id: roomId } });
  if (!room) return null;
  const senderId = room.createdByUserId;
  if (!senderId) return null;
  return createMessage(roomId, senderId, { messageType: "system", text });
}

export function formatMessage(message: any) {
  const reactions = Array.isArray(message.reactions)
    ? message.reactions.reduce((acc: Record<string, { count: number; users: Array<{ id: string; displayName?: string | null }> }>, reaction: any) => {
        const key = String(reaction.reaction ?? "");
        if (!key) return acc;
        acc[key] ??= { count: 0, users: [] };
        acc[key].count += 1;
        acc[key].users.push({ id: reaction.userId, displayName: reaction.user?.displayName ?? reaction.user?.username ?? null });
        return acc;
      }, {})
    : {};

  return {
    id: message.id,
    clientMessageId: message.clientMessageId,
    roomId: message.roomId,
    senderUserId: message.senderUserId,
    senderDisplayName: message.sender?.displayName ?? message.sender?.username ?? "Unknown",
    senderAvatarUrl: message.sender?.avatarUrl ?? null,
    messageType: message.messageType,
    text: message.text,
    createdAt: message.createdAt,
    editedAt: message.editedAt,
    deletedAt: message.deletedAt,
    reactions,
    sharePayload: parseJsonObject(message.sharePayloadJson, null)
  };
}

import { createHmac, timingSafeEqual } from "node:crypto";
import { config } from "../config.js";

export type SocialTokenPayload = {
  userId: string;
  username?: string | null;
  displayName?: string | null;
  exp: number;
};

function base64UrlEncode(value: string): string {
  return Buffer.from(value, "utf8").toString("base64url");
}

function base64UrlDecode(value: string): string {
  return Buffer.from(value, "base64url").toString("utf8");
}

function sign(data: string): string {
  return createHmac("sha256", config.socialTokenSecret).update(data).digest("base64url");
}

export function createSocialToken(payload: Omit<SocialTokenPayload, "exp">, ttlSeconds = 7 * 24 * 60 * 60): string {
  const fullPayload: SocialTokenPayload = {
    ...payload,
    exp: Math.floor(Date.now() / 1000) + ttlSeconds
  };
  const encodedPayload = base64UrlEncode(JSON.stringify(fullPayload));
  return `${encodedPayload}.${sign(encodedPayload)}`;
}

export function verifySocialToken(token: string): SocialTokenPayload | null {
  const [encodedPayload, signature] = token.split(".");
  if (!encodedPayload || !signature) return null;

  const expected = sign(encodedPayload);
  const expectedBuffer = Buffer.from(expected);
  const actualBuffer = Buffer.from(signature);
  if (expectedBuffer.length !== actualBuffer.length || !timingSafeEqual(expectedBuffer, actualBuffer)) return null;

  try {
    const payload = JSON.parse(base64UrlDecode(encodedPayload)) as SocialTokenPayload;
    if (!payload.userId || payload.exp < Math.floor(Date.now() / 1000)) return null;
    return payload;
  } catch {
    return null;
  }
}

import { config } from "../config.js";

async function tmdbFetch(
  path: string,
  params: Record<string, string | number | undefined> = {},
  method = "GET",
  body?: any
) {
  if (!config.tmdbApiKey) {
    throw new Error("TMDB API key is not configured on the server.");
  }
  const url = new URL(`https://api.themoviedb.org/3${path}`);
  for (const [key, value] of Object.entries(params)) {
    if (value != null && String(value).trim()) {
      url.searchParams.set(key, String(value));
    }
  }
  const headers: Record<string, string> = {
    accept: "application/json",
    "content-type": "application/json",
  };
  if (config.tmdbApiKey.startsWith("ey")) {
    headers.authorization = `Bearer ${config.tmdbApiKey}`;
  } else {
    url.searchParams.set("api_key", config.tmdbApiKey);
  }

  const response = await fetch(url.toString(), {
    method,
    headers,
    body: body ? JSON.stringify(body) : undefined,
  });

  if (!response.ok) {
    const errorText = await response.text().catch(() => "");
    throw new Error(`TMDb API error (${response.status}): ${errorText || response.statusText}`);
  }

  return response.json();
}

function imageUrl(path?: string | null, size = "w500"): string | null {
  return path ? `https://image.tmdb.org/t/p/${size}${path}` : null;
}

function yearFrom(value?: string | null): string | null {
  const match = String(value ?? "").match(/\b(19\d{2}|20\d{2})\b/);
  return match?.[1] ?? null;
}

function publicTmdbItem(record: Record<string, unknown>, mediaType: "movie" | "tv", sourceLabel: string) {
  return {
    id: String(record.id ?? ""),
    streamId: "",
    title: String(record.title ?? record.name ?? "Untitled"),
    type: mediaType === "tv" ? "series" : "movie",
    contentType: mediaType === "tv" ? "series" : "movie",
    year: yearFrom(String(record.release_date ?? record.first_air_date ?? "")),
    rating: typeof record.vote_average === "number" ? record.vote_average : null,
    description: String(record.overview ?? ""),
    coverUrl: imageUrl(String(record.poster_path ?? "")),
    backdropUrl: imageUrl(String(record.backdrop_path ?? ""), "w1280"),
    metadataProvider: "TMDb",
    metadataId: String(record.id ?? ""),
    tmdbId: String(record.id ?? ""),
    sourceLabel,
  };
}

export async function getTmdbCollection(
  collection: "popular" | "top_rated" | "trending",
  type: "movie" | "tv" | "all" = "movie",
  limit = 40
) {
  const mediaTypes: Array<"movie" | "tv"> = type === "all" ? ["movie", "tv"] : [type];
  const items: unknown[] = [];
  const maxItems = Math.max(1, Math.min(limit, 300));
  const pagesPerType = Math.max(1, Math.ceil(maxItems / mediaTypes.length / 20));
  for (const mediaType of mediaTypes) {
    const path = collection === "trending" ? `/trending/${mediaType}/day` : `/${mediaType}/${collection}`;
    for (let page = 1; page <= pagesPerType && items.length < maxItems; page += 1) {
      const data = await tmdbFetch(path, { language: "en-US", page }) as { results?: unknown[] };
      const results = Array.isArray(data.results) ? data.results : [];
      if (results.length === 0) break;
      items.push(...results.map((item) =>
        publicTmdbItem(
          item as Record<string, unknown>,
          mediaType,
          `TMDb ${collection.replace("_", " ")}`
        )
      ));
    }
  }
  return items.slice(0, maxItems);
}

export async function getTmdbAccountWatchlist(
  sessionId: string,
  accountId: string,
  type: "movie" | "tv" | "all" = "all",
  limit = 40
) {
  const mediaTypes: Array<"movie" | "tv"> = type === "all" ? ["movie", "tv"] : [type];
  const items: unknown[] = [];
  for (const mediaType of mediaTypes) {
    const data = await tmdbFetch(`/account/${accountId}/watchlist/${mediaType}`, {
      session_id: sessionId,
      language: "en-US",
      sort_by: "created_at.desc",
      page: 1,
    }) as { results?: unknown[] };
    const results = Array.isArray(data.results) ? data.results : [];
    items.push(...results.map((item) =>
      publicTmdbItem(item as Record<string, unknown>, mediaType, "TMDb Watchlist")
    ));
  }
  return items.slice(0, Math.max(1, Math.min(limit, 100)));
}

export type WebSearchResult = {

};

const cooldownByQuery = new Map<string, number>();

function cleanResult(result: Partial<WebSearchResult>, source: WebSearchResult["source"]): WebSearchResult | null {
  const title = String(result.title ?? "").trim();
  const url = String(result.url ?? "").trim();
  const snippet = String(result.snippet ?? "").trim();
  if (!title && !snippet) return null;
  return {
    title: title || "DuckDuckGo result",
    url,
    snippet,
    source
  };
}

async function searchWithDuckDuckScrape(query: string, maxResults: number): Promise<WebSearchResult[]> {
  const module = (await import("duck-duck-scrape")) as {
    search?: (query: string, options?: Record<string, unknown>) => Promise<{ results?: Array<Record<string, unknown>> }>;
  };
  if (!module.search) return [];
  const response = await module.search(query, { safeSearch: "STRICT" });
  return (response.results ?? [])
    .slice(0, maxResults)
    .map((result) =>
      cleanResult(
        {
          title: typeof result.title === "string" ? result.title : undefined,
          url: typeof result.url === "string" ? result.url : undefined,
          snippet: typeof result.description === "string"
            ? result.description
            : typeof result.snippet === "string"
              ? result.snippet
              : undefined
        },
        "duckduckgo"
      )
    )
    .filter((result): result is WebSearchResult => result != null);
}

async function searchWithInstantAnswer(query: string, maxResults: number): Promise<WebSearchResult[]> {
  const url = `https://api.duckduckgo.com/?q=${encodeURIComponent(query)}&format=json&no_html=1&skip_disambig=1`;
  const response = await fetch(url);
  if (!response.ok) return [];
  const data = (await response.json()) as {
    AbstractText?: string;
    AbstractURL?: string;
    Heading?: string;
    RelatedTopics?: Array<{ Text?: string; FirstURL?: string }>;
  };
  const results: WebSearchResult[] = [];
  const abstractResult = cleanResult(
    { title: data.Heading, url: data.AbstractURL, snippet: data.AbstractText },
    "instantAnswer"
  );
  if (abstractResult) results.push(abstractResult);
  for (const topic of data.RelatedTopics ?? []) {
    const result = cleanResult(
      { title: topic.Text?.split(" - ")[0], url: topic.FirstURL, snippet: topic.Text },
      "instantAnswer"
    );
    if (result) results.push(result);
  }
  return results.slice(0, maxResults);
}

export async function searchWeb(query: string, options: { maxResults?: number } = {}): Promise<WebSearchResult[]> {
  const normalizedQuery = query.trim().slice(0, 200);
  const maxResults = Math.max(1, Math.min(options.maxResults ?? 5, 10));
  if (!normalizedQuery) return [];

  try {
    const cached = await prisma.webSearchCache.findUnique({ where: { query: normalizedQuery } });
    if (cached && cached.expiresAt > new Date()) {
      return JSON.parse(cached.resultsJson) as WebSearchResult[];
    }
  } catch {
    return [];
  }

  const lastSearchAt = cooldownByQuery.get(normalizedQuery) ?? 0;
  if (Date.now() - lastSearchAt < cooldownMs) return [];
  cooldownByQuery.set(normalizedQuery, Date.now());

  let results: WebSearchResult[] = [];
  try {
    results = await searchWithDuckDuckScrape(normalizedQuery, maxResults);
  } catch {
    results = [];
  }

  if (results.length === 0) {
    try {
      results = await searchWithInstantAnswer(normalizedQuery, maxResults);
    } catch {
      results = [];
    }
  }

  try {
    await prisma.webSearchCache.upsert({
      where: { query: normalizedQuery },
      update: {
        resultsJson: JSON.stringify(results),
        expiresAt: new Date(Date.now() + cacheTtlMs)
      },
      create: {
        query: normalizedQuery,
        resultsJson: JSON.stringify(results),
        expiresAt: new Date(Date.now() + cacheTtlMs)
      }
    });
  } catch {
    return results;
  }

  return results;
}
